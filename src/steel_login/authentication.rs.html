5use reqwest::StatusCode;

9const MOJANG_AUTH_URL: &str =
10    "https://sessionserver.mojang.com/session/minecraft/hasJoined?username=";
11const SERVER_ID_ARG: &str = "&serverId=";
12
13/// An error that can occur during Mojang authentication.
14#[derive(Error, Debug)]
15pub enum AuthError {
16    /// Authentication servers are down.
17    #[error("Authentication servers are down")]
18    FailedResponse,
19    /// Failed to verify username.
20    #[error("Failed to verify username")]
21    UnverifiedUsername,
22    /// You are banned from Authentication servers.
23    #[error("You are banned from Authentication servers")]
24    Banned,
25    /// An error occurred with textures.
26    #[error("Texture Error {0}")]
27    TextureError(TextureError),
28    /// You have disallowed actions from Authentication servers.
29    #[error("You have disallowed actions from Authentication servers")]
30    DisallowedAction,
31    /// Failed to parse JSON into Game Profile.
32    #[error("Failed to parse JSON into Game Profile")]
33    FailedParse,
34    /// An unknown status code was returned.
35    #[error("Unknown Status Code {0}")]
36    UnknownStatusCode(StatusCode),
37}

39/// An error that can occur with textures.
40#[derive(Error, Debug)]
41pub enum TextureError {
42    /// Invalid URL.
43    #[error("Invalid URL")]
44    InvalidURL,
45    /// Invalid URL scheme for player texture.
46    #[error("Invalid URL scheme for player texture: {0}")]
47    DisallowedUrlScheme(String),
48    /// Invalid URL domain for player texture.
49    #[error("Invalid URL domain for player texture: {0}")]
50    DisallowedUrlDomain(String),
51    /// Failed to decode base64 player texture.
52    #[error("Failed to decode base64 player texture: {0}")]
53    DecodeError(String),
54    /// Failed to parse JSON from player texture.
55    #[error("Failed to parse JSON from player texture: {0}")]
56    JSONError(String),
57}

59const MAX_RETRIES: u32 = 3;
60
61/// Authenticates a player with Mojang's servers.
62pub async fn mojang_authenticate(
63    username: &str,
64    server_hash: &str,
65) -> Result<GameProfile, AuthError> {
66    let cap = MOJANG_AUTH_URL.len() + SERVER_ID_ARG.len() + username.len() + server_hash.len();
67    let mut auth_url = String::with_capacity(cap);
68    auth_url += MOJANG_AUTH_URL;
69    auth_url += username;
70    auth_url += SERVER_ID_ARG;
71    auth_url += server_hash;
72
73    let mut last_error = AuthError::FailedResponse;
74
75    for _ in 0..MAX_RETRIES {
76        let Ok(response) = reqwest::get(&auth_url).await else {
77            last_error = AuthError::FailedResponse;
78            continue;
79        };
80
81        match response.status() {
82            StatusCode::OK => {
83                return response.json().await.map_err(|_| AuthError::FailedParse);
84            }
85            StatusCode::NO_CONTENT => last_error = AuthError::UnverifiedUsername,
86            other => last_error = AuthError::UnknownStatusCode(other),
87        }
88    }
89
90    log::warn!("Player {username} auth failed");
91
92    Err(last_error)
93}

95/// Converts a signed bytes big endian to a hex string.
96///
97/// Equivalent to Java's `new BigInteger(bytes).toString(16)`.
98/// The first byte determines sign (two's complement). Leading zero bytes
99/// are not significant for magnitude but ARE significant for sign.
100#[must_use]
101pub fn signed_bytes_be_to_hex(bytes: &[u8]) -> String {
102    if bytes.is_empty() {
103        return "0".to_string();
104    }
105
106    // Sign is determined by the MSB of the first byte in the full array,
107    // NOT after stripping leading zeros. A leading 0x00 byte means the
108    // number is positive even if the next byte has its high bit set.
109    let is_negative = (bytes[0] & 0x80) != 0;
110
111    if is_negative {
112        // Negative case: calculate two's complement of the full byte array.
113        let mut magnitude: Vec<u8> = bytes.iter().map(|b| !*b).collect();
114        for byte in magnitude.iter_mut().rev() {
115            let (result, carry) = byte.overflowing_add(1);
116            *byte = result;
117            if !carry {
118                break;
119            }
120        }
121
122        let hex = hex::encode(&magnitude);
123        let trimmed = hex.trim_start_matches('0');
124        format!("-{trimmed}")
125    } else {
126        let hex = hex::encode(bytes);
127        let trimmed = hex.trim_start_matches('0');
128        if trimmed.is_empty() {
129            "0".to_string()
130        } else {
131            trimmed.to_string()
132        }
133    }
134}

136#[cfg(test)]
137mod tests {
138    use super::*;
139    use sha2::Digest;
140
141    #[test]
142    fn test_positive_simple() {
143        // BigInteger([0x01, 0x2a]).toString(16) = "12a"
144        assert_eq!(signed_bytes_be_to_hex(&[0x01, 0x2a]), "12a");
145    }
146
147    #[test]
148    fn test_negative_simple() {
149        // BigInteger([0xff]).toString(16) = "-1"
150        assert_eq!(signed_bytes_be_to_hex(&[0xff]), "-1");
151    }
152
153    #[test]
154    fn test_leading_zero_preserves_positive_sign() {
155        // This was the bug: [0x00, 0x9a] is positive because the first byte is 0x00.
156        // BigInteger([0x00, 0x9a]).toString(16) = "9a"
157        assert_eq!(signed_bytes_be_to_hex(&[0x00, 0x9a]), "9a");
158    }
159
160    #[test]
161    fn test_leading_zero_with_high_bit() {
162        // BigInteger([0x00, 0xff, 0xab]).toString(16) = "ffab"
163        assert_eq!(signed_bytes_be_to_hex(&[0x00, 0xff, 0xab]), "ffab");
164    }
165
166    #[test]
167    fn test_negative_twos_complement() {
168        // BigInteger([0x80]).toString(16) = "-80"
169        assert_eq!(signed_bytes_be_to_hex(&[0x80]), "-80");
170    }
171
172    #[test]
173    fn test_negative_multi_byte() {
174        // BigInteger([0xfe, 0xdc]).toString(16) = "-124"
175        assert_eq!(signed_bytes_be_to_hex(&[0xfe, 0xdc]), "-124");
176    }
177
178    #[test]
179    fn test_zero() {
180        assert_eq!(signed_bytes_be_to_hex(&[0x00]), "0");
181        assert_eq!(signed_bytes_be_to_hex(&[0x00, 0x00, 0x00]), "0");
182    }
183
184    #[test]
185    fn test_empty() {
186        assert_eq!(signed_bytes_be_to_hex(&[]), "0");
187    }
188
189    #[test]
190    fn test_known_notchian_hashes() {
191        // Known test vectors from wiki.vg
192        assert_eq!(
193            signed_bytes_be_to_hex(&sha1::Sha1::digest(b"Notch")),
194            "4ed1f46bbe04bc756bcb17c0c7ce3e4632f06a48"
195        );
196        assert_eq!(
197            signed_bytes_be_to_hex(&sha1::Sha1::digest(b"jeb_")),
198            "-7c9d5b0044c130109a5d7b5fb5c317c02b4e28c1"
199        );
200        assert_eq!(
201            signed_bytes_be_to_hex(&sha1::Sha1::digest(b"simon")),
202            "88e16a1019277b15d58faf0541e11910eb756f6"
203        );
204    }
