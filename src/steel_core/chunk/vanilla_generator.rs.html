7/// A chunk generator for vanilla (normal) world generation.
8///
9/// Matches vanilla's `NoiseBasedChunkGenerator`. The biome source is pluggable
10/// per-dimension — overworld, nether, and end each provide a different
11/// [`BiomeSourceKind`] variant.
12pub struct VanillaGenerator {
13    /// Biome source for this dimension. Determines biomes at each quart position.
14    biome_source: BiomeSourceKind,
15}
16
17impl VanillaGenerator {
18    /// Creates a new `VanillaGenerator` with the given biome source.
19    #[must_use]
20    pub const fn new(biome_source: BiomeSourceKind) -> Self {
21        Self { biome_source }
22    }
23}

25impl ChunkGenerator for VanillaGenerator {
26    fn create_structures(&self, _chunk: &ChunkAccess) {}
27
28    fn create_biomes(&self, chunk: &ChunkAccess) {
29        let pos = chunk.pos();
30        let min_y = chunk.min_y();
31        let section_count = chunk.sections().sections.len();
32
33        let chunk_x = pos.0.x;
34        let chunk_z = pos.0.y;
35
36        let mut sampler = self.biome_source.chunk_sampler();
37
38        // Column-major iteration: sample all Y values for each (X, Z) column
39        // before moving to the next column. This keeps the column cache effective —
40        // column-level density functions (continents, erosion, ridges, etc.) are
41        // computed once per column instead of once per sample.
42        for local_quart_x in 0..4i32 {
43            for local_quart_z in 0..4i32 {
44                let quart_x = chunk_x * 4 + local_quart_x;
45                let quart_z = chunk_z * 4 + local_quart_z;
46
47                for section_index in 0..section_count {
48                    let section_y = (min_y / 16) + section_index as i32;
49                    let section = &chunk.sections().sections[section_index];
50                    let mut section_guard = section.write();
51
52                    for local_quart_y in 0..4i32 {
53                        let quart_y = section_y * 4 + local_quart_y;
54
55                        let biome = sampler.sample(quart_x, quart_y, quart_z);
56                        let biome_id = *REGISTRY.biomes.get_id(biome) as u16;
57
58                        section_guard.biomes.set(
59                            local_quart_x as usize,
60                            local_quart_y as usize,
61                            local_quart_z as usize,
62                            biome_id,
63                        );
64                    }
65                }
66            }
67        }
68
69        chunk.mark_dirty();
70    }
71
72    fn fill_from_noise(&self, chunk: &ChunkAccess) {
73        // TODO: Implement actual noise-based terrain generation (NoiseChunk + trilinear interpolation)

80                    REGISTRY

82                        .get_default_state_id(vanilla_blocks::GRASS_BLOCK),
83                );
84            }
85        }
86    }
