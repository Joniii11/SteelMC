8/// A map of radius to the required chunk status.
9#[derive(Debug, Clone)]
10pub struct ChunkDependencies {
11    dependency_by_radius: Box<[ChunkStatus]>,
12    radius_by_dependency: Box<[usize]>,
13}
14
15impl ChunkDependencies {
16    /// Creates a new chunk dependencies.
17    #[must_use]
18    pub fn new(dependency_by_radius: Vec<ChunkStatus>) -> Self {
19        let dependency_by_radius = dependency_by_radius.into_boxed_slice();
20        let size = dependency_by_radius
21            .first()
22            .map_or(0, |s| s.get_index() + 1);
23        let mut radius_by_dependency = vec![0; size].into_boxed_slice();
24
25        for (radius, dependency) in dependency_by_radius.iter().enumerate() {
26            let index = dependency.get_index();
27            for i in 0..=index {
28                if i < radius_by_dependency.len() {
29                    radius_by_dependency[i] = radius;
30                }
31            }
32        }
33
34        Self {
35            dependency_by_radius,
36            radius_by_dependency,
37        }
38    }
39
40    /// Gets the radius of the dependencies for the given status.
41    #[must_use]
42    pub fn get_radius_of(&self, status: ChunkStatus) -> usize {
43        let index = status.get_index();
44        if index >= self.radius_by_dependency.len() {
45            // This would be an error in Java, but Rust prefers safe access.
46            // Returning the max radius is a possible fallback.
47            return self.get_radius();
48        }
49        self.radius_by_dependency[index]
50    }
51
52    /// Gets the radius of the dependencies.
53    #[must_use]
54    pub fn get_radius(&self) -> usize {
55        self.dependency_by_radius.len().saturating_sub(1)
56    }
57
58    /// Gets the dependencies for the given distance.
59    #[must_use]
60    pub fn get(&self, distance: usize) -> Option<ChunkStatus> {
61        self.dependency_by_radius.get(distance).copied()
62    }
63}

91/// A builder for a chunk step.
92pub struct Builder {
93    status: ChunkStatus,
94    parent: Option<Arc<ChunkStep>>,
95    direct_dependencies_by_radius: Vec<ChunkStatus>,
96    block_state_write_radius: i32,
97    task: ChunkStatusTask,
98}
99
100impl Builder {
101    #[must_use]
102    /// Creates a new chunk step builder.
103    ///
104    /// # Panics
105    ///
106    /// Panics if the status is not the next status of the parent.
107    pub fn new(status: ChunkStatus, parent: Option<Arc<ChunkStep>>) -> Self {
108        let direct_dependencies_by_radius = if let Some(p) = &parent {
109            assert!(
110                p.target_status.next() == Some(status),
111                "Out of order status: {:?}, expected next of {:?}",
112                status,
113                p.target_status
114            );
115            vec![p.target_status]
116        } else {
117            assert!(
118                status.parent().is_none(),
119                "Not starting with the first status: {status:?}"
120            );
121            Vec::new()
122        };
123
124        Self {
125            status,
126            parent,
127            direct_dependencies_by_radius,
128            block_state_write_radius: -1,
129            task: || {},
130        }
131    }
132
133    #[must_use]
134    /// Adds a requirement to the step.
135    ///
136    /// # Panics
137    ///
138    /// Panics if the status is greater than the current status.
139    pub fn add_requirement(mut self, status: ChunkStatus, radius: usize) -> Self {
140        assert!(
141            status < self.status,
142            "Status {:?} can not be required by {:?}",
143            status,
144            self.status
145        );
146
147        let new_len = radius + 1;
148        let old_len = self.direct_dependencies_by_radius.len();
149
150        if new_len > old_len {
151            self.direct_dependencies_by_radius.resize(new_len, status);
152        }
153
154        for i in 0..old_len.min(new_len) {
155            self.direct_dependencies_by_radius[i] =
156                self.direct_dependencies_by_radius[i].max(status);
157        }
158        self
159    }
160
161    #[must_use]
162    /// Sets the block state write radius for the step.
163    pub fn block_state_write_radius(mut self, radius: i32) -> Self {
164        self.block_state_write_radius = radius;
165        self
166    }
167
168    /// Sets the task for the step.
169    #[must_use]
170    pub fn set_task(mut self, task: ChunkStatusTask) -> Self {
171        self.task = task;
172        self
173    }
174
175    #[must_use]
176    /// Builds the chunk step.
177    pub fn build(self) -> ChunkStep {
178        let accumulated_dependencies = self.build_accumulated_dependencies();
179        let direct_dependencies = ChunkDependencies::new(self.direct_dependencies_by_radius);
180
181        ChunkStep {
182            target_status: self.status,
183            direct_dependencies: Arc::new(direct_dependencies),
184            accumulated_dependencies: Arc::new(accumulated_dependencies),
185            block_state_write_radius: self.block_state_write_radius,
186            task: self.task,
187        }
188    }
189
190    fn build_accumulated_dependencies(&self) -> ChunkDependencies {
191        if self.parent.is_none() {
192            return ChunkDependencies::new(self.direct_dependencies_by_radius.clone());
193        }
194        let parent = self.parent.as_ref().expect("Parent is required");
195
196        let radius_of_parent = self
197            .direct_dependencies_by_radius
198            .iter()
199            .rposition(|&s| s >= parent.target_status)
200            .unwrap_or(0);
201
202        let parent_deps = &parent.accumulated_dependencies;
203        let new_len = max(
204            radius_of_parent + parent_deps.dependency_by_radius.len(),
205            self.direct_dependencies_by_radius.len(),
206        );
207        let mut accumulated = Vec::with_capacity(new_len);
208
209        for dist in 0..new_len {
210            let dist_in_parent = dist.saturating_sub(radius_of_parent);
211
212            let parent_dep = parent_deps.get(dist_in_parent);
213            let direct_dep = self.direct_dependencies_by_radius.get(dist).copied();
214
215            let dep = match (direct_dep, parent_dep) {
216                (Some(d), Some(p)) => d.max(p),
217                (Some(d), None) => d,
218                (None, Some(p)) => p,
219                (None, None) => continue,
220            };
221            accumulated.push(dep);
222        }
223
224        ChunkDependencies::new(accumulated)
225    }
226}

228/// Represents the hierarchy and dependencies for chunk generation or loading.
229pub struct ChunkPyramid {
230    steps: Box<[Arc<ChunkStep>]>,
231}
232
233impl ChunkPyramid {
234    /// Gets the step for the given status.
235    #[must_use]
236    pub fn get_step_to(&self, status: ChunkStatus) -> Arc<ChunkStep> {
237        self.steps[status.get_index()].clone()
238    }
239}

241/// A builder for a chunk pyramid.
242pub struct ChunkPyramidBuilder {
243    steps: Vec<Arc<ChunkStep>>,
244}
245
246impl Default for ChunkPyramidBuilder {
247    fn default() -> Self {
248        Self::new()
249    }
250}
251
252impl ChunkPyramidBuilder {
253    #[must_use]
254    /// Creates a new chunk pyramid builder.
255    pub fn new() -> Self {
256        Self { steps: Vec::new() }
257    }
258
259    /// Adds a new step to the chunk pyramid.
260    #[must_use]
261    pub fn step<F>(mut self, status: ChunkStatus, op: F) -> Self
262    where
263        F: FnOnce(Builder) -> Builder,
264    {
265        let parent = self.steps.last().cloned();
266        let step_builder = ChunkStep::builder(status, parent);
267        let built_step = op(step_builder).build();
268        self.steps.push(Arc::new(built_step));
269        self
270    }
271
272    #[must_use]
273    /// Builds the chunk pyramid.
274    pub fn build(self) -> ChunkPyramid {
275        ChunkPyramid {
276            steps: self.steps.into_boxed_slice(),
277        }
278    }
279}
