1//! This module contains the `ChunkMap` struct, which is a map of chunk holders.
2use std::sync::Arc;
3use steel_utils::ChunkPos;
4use tokio::sync::Mutex;
5use tokio_util::task::TaskTracker;
6
7use crate::chunk::chunk_holder::ChunkHolder;
8use crate::chunk::{chunk_access::ChunkStatus, chunk_generation_task::ChunkGenerationTask};
9
10/// A map of chunks.
11pub struct ChunkMap {
12    /// A map of all the chunks in the level.
13    pub chunks: scc::HashMap<ChunkPos, Arc<ChunkHolder>>,
14    /// A queue of pending generation tasks.
15    pub pending_generation_tasks: Mutex<Vec<Arc<ChunkGenerationTask>>>,
16    /// A tracker for the generation tasks.
17    pub task_tracker: TaskTracker,
18}
19
20impl Default for ChunkMap {
21    fn default() -> Self {
22        Self::new()
23    }
24}
25
26impl ChunkMap {
27    /// Creates a new chunk map.
28    #[must_use]
29    pub fn new() -> Self {
30        Self {
31            chunks: scc::HashMap::new(),
32            pending_generation_tasks: Mutex::new(Vec::new()),
33            task_tracker: TaskTracker::new(),
34        }
35    }
36
37    /// Schedules a new generation task for the given position and target status.
38    ///
39    /// Returns a handle to the task.
40    pub async fn schedule_generation_task(
41        self: &Arc<Self>,
42        target_status: ChunkStatus,
43        pos: ChunkPos,
44    ) -> Arc<ChunkGenerationTask> {
45        let task = Arc::new(ChunkGenerationTask::new(pos, target_status, self.clone()).await);
46        self.pending_generation_tasks
47            .lock()
48            .await
49            .push(task.clone());
50        task
51    }
52
53    /// Runs the generation tasks.
54    pub async fn run_generation_tasks(&self) {
55        let mut pending_generation_tasks = self.pending_generation_tasks.lock().await;
56        for task in pending_generation_tasks.drain(..) {
57            self.task_tracker.spawn(async move {
58                task.run().await;
59            });
60        }
61    }
62}
