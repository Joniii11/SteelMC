1//! End islands terrain generation algorithm.
2//!
3//! Matches vanilla's `DensityFunctions.EndIslandDensityFunction`. Generates the
4//! characteristic floating island pattern of The End by combining a distance-based
5//! falloff from the origin with simplex-noise-driven island placement.
6//!
7//! The noise seed is always 0 (world-seed-independent), initialized with
8//! `LegacyRandomSource(0)` + `consumeCount(17292)`.
9//!
10//! Result range: `[-0.84375, 0.5625]`.
11
12use steel_utils::noise::SimplexNoise;
13use steel_utils::random::Random;
14use steel_utils::random::legacy_random::LegacyRandom;
15
16/// Threshold for simplex noise below which an island is spawned.
17///
18/// Vanilla uses `-0.9F` (float literal) in a `double < float` comparison, which
19/// promotes the float to double. `(double)(-0.9f)` ≈ `-0.8999999761581421`,
20/// NOT the exact double `-0.9`. We must match this f32→f64 promotion.
21const ISLAND_THRESHOLD: f64 = -0.9_f32 as f64;
22
23/// End islands density function.
24///
25/// Unlike overworld/nether density functions which are transpiled into native Rust,
26/// this is used directly at runtime because it's a self-contained leaf algorithm
27/// (simplex noise + neighbor loop) with no density function tree to transpile.
28#[derive(Debug, Clone)]
29pub struct EndIslands {
30    island_noise: SimplexNoise,
31}
32
33impl EndIslands {
34    /// Create a new `EndIslands` with the given world seed.
35    ///
36    /// Matches vanilla's `RandomState.NoiseWiringHelper.wrapNew()` which creates
37    /// `EndIslandDensityFunction(worldSeed)`, NOT seed 0. The JSON codec defaults
38    /// to seed 0, but `RandomState` replaces it with the world seed.
39    #[must_use]
40    pub fn new(seed: u64) -> Self {
41        let mut rng = LegacyRandom::from_seed(seed);
42        rng.consume_count(17292);
43        let island_noise = SimplexNoise::new(&mut rng);
44        Self { island_noise }
45    }
46
47    /// Sample the erosion value at block coordinates.
48    ///
49    /// Used by `EndBiomeSource` to determine biome thresholds. Converts block
50    /// coordinates to section coordinates internally (divides by 8).
51    #[must_use]
52    pub fn sample(&self, block_x: i32, _block_y: i32, block_z: i32) -> f64 {
53        // Widen to f64 BEFORE subtracting 8.0, matching Java's `float - 8.0` (double literal)
54        // where the float is promoted to double first.
55        (f64::from(Self::get_height_value(
56            &self.island_noise,
57            block_x / 8,
58            block_z / 8,
59        )) - 8.0)
60            / 128.0
61    }
62
63    /// Compute the height value at section coordinates.
64    ///
65    /// Matches vanilla's `EndIslandDensityFunction.getHeightValue()`.
66    /// Takes section coordinates (block position / 8).
67    fn get_height_value(island_noise: &SimplexNoise, section_x: i32, section_z: i32) -> f32 {
68        let chunk_x = section_x / 2;
69        let chunk_z = section_z / 2;
70        let sub_section_x = section_x % 2;
71        let sub_section_z = section_z % 2;
72
73        // Distance-based falloff from the origin.
74        // Vanilla does integer multiply THEN casts to float: `Mth.sqrt(sectionX * sectionX + ...)`.
75        // Integer overflow wraps in Java; we use wrapping_mul/wrapping_add to match.
76        let dist_sq = section_x
77            .wrapping_mul(section_x)
78            .wrapping_add(section_z.wrapping_mul(section_z));
79        let dist = (dist_sq as f32).sqrt();
80        let mut doffs = (100.0_f32 - dist * 8.0).clamp(-100.0, 80.0);
81
82        // Check 25×25 neighborhood for island contributions
83        for xo in -12..=12 {
84            for zo in -12..=12 {
85                let total_chunk_x = i64::from(chunk_x) + i64::from(xo);
86                let total_chunk_z = i64::from(chunk_z) + i64::from(zo);
87
88                if total_chunk_x * total_chunk_x + total_chunk_z * total_chunk_z > 4096
89                    && island_noise.get_value_2d(total_chunk_x as f64, total_chunk_z as f64)
90                        < ISLAND_THRESHOLD
91                {
92                    let island_size = ((total_chunk_x as f32).abs() * 3439.0
93                        + (total_chunk_z as f32).abs() * 147.0)
94                        % 13.0
95                        + 9.0;
96                    let xd = sub_section_x as f32 - (xo * 2) as f32;
97                    let zd = sub_section_z as f32 - (zo * 2) as f32;
98                    let new_doffs =
99                        (100.0_f32 - (xd * xd + zd * zd).sqrt() * island_size).clamp(-100.0, 80.0);
100                    doffs = doffs.max(new_doffs);
101                }
102            }
103        }
104
105        doffs
106    }
107}
