1//! Chat and messaging state for a player.
2//!
3//! Groups the fields related to secure chat: message counters, signature cache,
4//! message validator, chat session, and message chain.
5
6use super::LastSeenMessagesValidator;
7use super::MessageCache;
8use super::message_chain::SignedMessageChain;
9use super::profile_key::RemoteChatSession;
10
11/// All chat-related state for a player.
12///
13/// Stored behind a single `SyncMutex` on `Player`. The fields were previously
14/// individual atomics/mutexes but are always accessed within short critical
15/// sections per-player, so a single lock is simpler with no real contention cost.
16pub struct ChatState {
17    /// Counter for chat messages sent BY this player.
18    pub messages_sent: i32,
19    /// Counter for chat messages received BY this player.
20    pub messages_received: i32,
21    /// Message signature cache for tracking chat messages.
22    pub signature_cache: MessageCache,
23    /// Validator for client acknowledgements of messages we've sent.
24    pub message_validator: LastSeenMessagesValidator,
25    /// Remote chat session containing the player's public key (if signed chat is enabled).
26    pub chat_session: Option<RemoteChatSession>,
27    /// Message chain state for tracking signed message sequence.
28    pub message_chain: Option<SignedMessageChain>,
29}
30
31impl ChatState {
32    pub fn new() -> Self {
33        Self {
34            messages_sent: 0,
35            messages_received: 0,
36            signature_cache: MessageCache::new(),
37            message_validator: LastSeenMessagesValidator::new(),
38            chat_session: None,
39            message_chain: None,
40        }
41    }
42}
