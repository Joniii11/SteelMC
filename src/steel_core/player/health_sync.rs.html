1//! Delta-tracking state for `CSetHealth` packet deduplication.
2//!
3//! Tracks the last health/food/saturation values sent to the client so we only
4//! send `CSetHealth` when something actually changes.
5//!
6//! Vanilla: `ServerPlayer.lastSentHealth`, `lastSentFood`, `lastFoodSaturationZero`.
7
8/// Tracks the last health/food/saturation values sent to the client.
9pub struct HealthSyncState {
10    /// Last health value sent to the client.
11    pub last_health: f32,
12    /// Last food level sent to the client.
13    pub last_food: i32,
14    /// Whether saturation was zero last time we sent health.
15    pub saturation_zero: bool,
16}
17
18impl HealthSyncState {
19    /// Creates a new state that will trigger a send on the first tick.
20    #[must_use]
21    pub const fn new() -> Self {
22        Self {
23            last_health: -1.0,
24            last_food: -1,
25            saturation_zero: true,
26        }
27    }
28
29    /// Returns true if the given values differ from what was last sent.
30    #[allow(clippy::float_cmp)]
31    #[must_use]
32    pub fn needs_update(&self, health: f32, food: i32, saturation_zero: bool) -> bool {
33        self.last_health != health
34            || self.last_food != food
35            || self.saturation_zero != saturation_zero
36    }
37
38    /// Records that we just sent the given values to the client.
39    pub const fn record_sent(&mut self, health: f32, food: i32, saturation_zero: bool) {
40        self.last_health = health;
41        self.last_food = food;
42        self.saturation_zero = saturation_zero;
43    }
44
45    /// Invalidates the state so the next tick will re-send.
46    ///
47    /// Vanilla: `resetSentInfo`.
48    pub const fn invalidate(&mut self) {
49        self.last_health = -1.0e8;
50    }
51
52    /// Resets to respawn defaults (forces re-send on next tick).
53    pub const fn reset_for_respawn(&mut self) {
54        self.last_health = -1.0;
55        self.last_food = -1;
56    }
57}
