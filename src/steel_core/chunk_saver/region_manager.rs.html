45/// Manages region files with seek-based chunk access.
46///
47/// Only keeps region headers (8KB each) in memory, not chunk data.
48/// Chunks are loaded on-demand and converted directly to runtime format.
49pub struct RegionManager {
50    /// Base directory for region files (e.g., "world/region").
51    base_path: PathBuf,
52    /// Open region file handles with their headers.
53    regions: AsyncRwLock<FxHashMap<RegionPos, RegionHandle>>,
54}

95    /// Ensures a block state exists in the chunk's palette, returning its index.
96    fn ensure_block_state(&mut self, block_id: BlockStateId) -> u16 {
97        // Get block and properties from registry
98        let block = self
99            .registry
100            .blocks
101            .by_state_id(block_id)
102            .expect("Invalid block state ID");
103        let properties = self.registry.blocks.get_properties(block_id);
104
105        let persistent = PersistentBlockState {
106            name: block.key.clone(),
107            properties: properties
108                .into_iter()
109                .map(|(k, v)| (k.to_string(), v.to_string()))
110                .collect(),
111        };
112
113        // Check if already exists
114        if let Some(idx) = self.block_states.iter().position(|s| s == &persistent) {
115            return idx as u16;
116        }
117
118        // Add new entry
119        let idx = self.block_states.len();
120        self.block_states.push(persistent);
121        idx as u16
122    }
123
124    /// Ensures a biome exists in the chunk's palette, returning its index.
125    fn ensure_biome(&mut self, biome_id: u8) -> u16 {
126        // Get biome identifier from registry
127        let biome = self
128            .registry
129            .biomes
130            .by_id(biome_id as usize)
131            .expect("Invalid biome ID");
132        let identifier = biome.key.clone();
133
134        if let Some(idx) = self.biomes.iter().position(|b| b == &identifier) {
135            return idx as u16;
136        }
137
138        let idx = self.biomes.len();
139        self.biomes.push(identifier);
140        idx as u16
141    }

147    /// # Arguments
148    /// * `base_path` - Directory where region files are stored.
149    /// * `registry` - The registry for block state and biome conversions.
150    pub fn new(base_path: impl Into<PathBuf>) -> Self {
151        Self {
152            base_path: base_path.into(),
153            regions: AsyncRwLock::new(FxHashMap::default()),
154        }

157    /// Gets the file path for a region.
158    fn region_path(&self, pos: RegionPos) -> PathBuf {
159        self.base_path.join(pos.filename())
160    }

162    /// Opens or creates a region file, loading only the header.
163    async fn open_region(&self, pos: RegionPos) -> io::Result<RegionHandle> {
164        let path = self.region_path(pos);
165
166        if !path.exists() {
167            // Create new region file with empty header
168            return self.create_region(pos).await;
169        }
170
171        let mut file = OpenOptions::new()
172            .read(true)
173            .write(true)
174            .open(&path)
175            .await?;
176
177        // Read and verify magic + version
178        let mut header_bytes = [0u8; FILE_HEADER_SIZE];
179        file.read_exact(&mut header_bytes).await?;
180
181        let magic = &header_bytes[0..4];
182        if magic != REGION_MAGIC {
183            return Err(io::Error::new(
184                io::ErrorKind::InvalidData,
185                "Invalid region file magic",
186            ));
187        }
188
189        let version = u16::from_le_bytes([header_bytes[4], header_bytes[5]]);
190        if version > FORMAT_VERSION {
191            return Err(io::Error::new(
192                io::ErrorKind::InvalidData,
193                format!(
194                    "Region file version {version} is newer than supported version {FORMAT_VERSION}"
195                ),
196            ));
197        }
198
199        // Read chunk table
200        let mut table_bytes = vec![0u8; CHUNK_TABLE_SIZE];
201        file.read_exact(&mut table_bytes).await?;
202        let header = RegionHeader::from_bytes(&table_bytes);
203
204        // Calculate file size in sectors
205        let file_size = file.seek(io::SeekFrom::End(0)).await?;
206        let file_sectors = file_size.div_ceil(SECTOR_SIZE as u64) as u32;
207
208        Ok(RegionHandle {
209            file,
210            header,
211            loaded_chunk_count: 0,
212            header_dirty: false,
213            file_sectors,
214        })
215    }
216
217    /// Creates a new empty region file.
218    async fn create_region(&self, pos: RegionPos) -> io::Result<RegionHandle> {
219        fs::create_dir_all(&self.base_path).await?;
220
221        let path = self.region_path(pos);
222        let mut file = OpenOptions::new()
223            .read(true)
224            .write(true)
225            .create(true)
226            .truncate(true)
227            .open(&path)
228            .await?;
229
230        // Write header
231        let mut header_bytes = [0u8; FILE_HEADER_SIZE];
232        header_bytes[0..4].copy_from_slice(&REGION_MAGIC);
233        header_bytes[4..6].copy_from_slice(&FORMAT_VERSION.to_le_bytes());
234        file.write_all(&header_bytes).await?;
235
236        // Write empty chunk table
237        let header = RegionHeader::new();
238        file.write_all(&header.to_bytes()).await?;
239        file.flush().await?;
240
241        Ok(RegionHandle {
242            file,
243            header,
244            loaded_chunk_count: 0,
245            header_dirty: false,
246            file_sectors: FIRST_DATA_SECTOR,
247        })
248    }

250    /// Writes the header to disk.
251    async fn write_header(file: &mut File, header: &RegionHeader) -> io::Result<()> {
252        file.seek(io::SeekFrom::Start(FILE_HEADER_SIZE as u64))
253            .await?;
254        file.write_all(&header.to_bytes()).await?;
255        file.flush().await?;
256        Ok(())
257    }

259    /// Reads a chunk's compressed data from disk.
260    async fn read_chunk_data(
261        file: &mut File,
262        sector_offset: u32,
263        size: u32,
264    ) -> io::Result<Vec<u8>> {
265        let byte_offset = u64::from(sector_offset) * SECTOR_SIZE as u64;
266        file.seek(io::SeekFrom::Start(byte_offset)).await?;
267
268        let mut compressed = vec![0u8; size as usize];
269        file.read_exact(&mut compressed).await?;
270        Ok(compressed)
271    }

273    /// Writes chunk data to disk at the specified sector offset.
274    async fn write_chunk_data(
275        file: &mut File,
276        sector_offset: u32,
277        data: &[u8],
278        file_sectors: &mut u32,
279    ) -> io::Result<()> {
280        let byte_offset = u64::from(sector_offset) * SECTOR_SIZE as u64;
281        file.seek(io::SeekFrom::Start(byte_offset)).await?;
282        file.write_all(data).await?;
283
284        // Pad to sector boundary
285        let padding_needed = (SECTOR_SIZE - (data.len() % SECTOR_SIZE)) % SECTOR_SIZE;
286        if padding_needed > 0 {
287            file.write_all(&vec![0u8; padding_needed]).await?;
288        }
289
290        // Update file sectors if we wrote past the end
291        let sectors_used = data.len().div_ceil(SECTOR_SIZE) as u32;
292        let end_sector = sector_offset + sectors_used;
293        if end_sector > *file_sectors {
294            *file_sectors = end_sector;
295        }
296
297        file.flush().await?;
298        Ok(())
299    }

301    /// Saves a chunk to the appropriate region.
302    ///
303    /// The chunk is serialized, compressed, and written to disk immediately.
304    /// If the region was already open (has loaded chunks), the header update is
305    /// deferred. If this call opened the region, it will be closed after saving.
306    ///
307    /// If the chunk is not dirty, this is a no-op and returns `Ok(false)`.
308    /// Returns `Ok(true)` if the chunk was saved.
309    /// Prepares chunk data for saving. Call this while holding the chunk lock,
310    /// then pass the result to `save_chunk_data` after releasing the lock.
311    #[must_use]
312    pub fn prepare_chunk_save(chunk: &ChunkAccess) -> Option<PreparedChunkSave> {
313        if !chunk.is_dirty() {
314            return None;
315        }
316
317        let pos = chunk.pos();
318
319        // Get block entities if this is a full chunk
320        let block_entities: Vec<SharedBlockEntity> = chunk
321            .as_full()
322            .map(super::super::chunk::level_chunk::LevelChunk::get_block_entities)
323            .unwrap_or_default();
324
325        // Get saveable entities if this is a full chunk
326        let entities: Vec<SharedEntity> = chunk
327            .as_full()
328            .map(|c| c.entities.get_saveable_entities())
329            .unwrap_or_default();
330
331        let persistent = Self::to_persistent(chunk.sections(), &block_entities, &entities, pos);
332
333        Some(PreparedChunkSave { pos, persistent })
334    }

336    /// Saves prepared chunk data to disk. This is the async part that doesn't
337    /// need to hold the chunk lock.
338    #[allow(clippy::missing_panics_doc)]
339    pub async fn save_chunk_data(
340        &self,
341        prepared: PreparedChunkSave,
342        status: ChunkStatus,
343    ) -> io::Result<bool> {
344        let pos = prepared.pos;
345        let region_pos = RegionPos::from_chunk(pos.0.x, pos.0.y);
346        let (local_x, local_z) = RegionPos::local_chunk_pos(pos.0.x, pos.0.y);
347        let index = RegionHeader::chunk_index(local_x, local_z);
348
349        let mut regions = self.regions.write().await;
350
351        // Track if we opened the region (so we can close it after)
352        let we_opened_region = !regions.contains_key(&region_pos);
353
354        // Get or open the region
355        let handle = if let Some(handle) = regions.get_mut(&region_pos) {
356            handle
357        } else {
358            let handle = self.open_region(region_pos).await?;
359            regions.insert(region_pos, handle);
360            regions.get_mut(&region_pos).expect("just inserted")
361        };
362
363        // Serialize the prepared data
364        let persistent = prepared.persistent;
365        let data = wincode::serialize(&persistent)
366            .map_err(|e| io::Error::new(io::ErrorKind::InvalidData, e.to_string()))?;
367
368        // Compress with zstd
369        let compressed = zstd::encode_all(&data[..], 3)?;
370
371        if compressed.len() > MAX_CHUNK_SIZE {
372            // Clean up if we opened the region
373            if we_opened_region {
374                regions.remove(&region_pos);
375            }
376            return Err(io::Error::new(
377                io::ErrorKind::InvalidData,
378                format!(
379                    "Chunk too large: {} bytes (max {})",
380                    compressed.len(),
381                    MAX_CHUNK_SIZE
382                ),
383            ));
384        }
385
386        // Find space for the chunk
387        let sectors_needed = compressed.len().div_ceil(SECTOR_SIZE) as u32;
388        let old_entry = handle.header.entries[index];
389
390        // Try to reuse existing space if it fits
391        let sector_offset = if old_entry.exists() && old_entry.sector_count() >= sectors_needed {
392            old_entry.sector_offset
393        } else {
394            handle
395                .header
396                .find_free_sectors(sectors_needed, handle.file_sectors)
397        };
398
399        // Write chunk data
400        Self::write_chunk_data(
401            &mut handle.file,
402            sector_offset,
403            &compressed,
404            &mut handle.file_sectors,
405        )
406        .await?;
407
408        // Update header entry
409        handle.header.entries[index] =
410            super::format::ChunkEntry::new(sector_offset, compressed.len() as u32, status);
411
412        // If we opened this region and no chunks are loaded from it,
413        // write the header and close it immediately
414        if we_opened_region && handle.loaded_chunk_count == 0 {
415            Self::write_header(&mut handle.file, &handle.header).await?;
416            regions.remove(&region_pos);
417        } else {
418            handle.header_dirty = true;
419        }
420
421        Ok(true)
422    }

424    /// Loads a chunk from the appropriate region.
425    ///
426    /// Automatically opens the region if not already open. The region's reference
427    /// count is incremented, so you must call `release_chunk` when done with the chunk.
428    ///
429    /// Returns `Ok(None)` if the chunk doesn't exist on disk.
430    ///
431    /// # Arguments
432    /// * `pos` - The chunk position
433    /// * `min_y` - The minimum Y coordinate of the world
434    /// * `height` - The total height of the world
435    /// * `level` - Weak reference to the world for `LevelChunk`
436    ///
437    /// The region must already be acquired via `acquire_chunk` before calling this.
438    #[allow(clippy::missing_panics_doc)]
439    pub async fn load_chunk(
440        &self,
441        pos: ChunkPos,
442        min_y: i32,
443        height: i32,
444        level: Weak<World>,
445    ) -> io::Result<Option<(ChunkAccess, ChunkStatus)>> {
446        let region_pos = RegionPos::from_chunk(pos.0.x, pos.0.y);
447        let (local_x, local_z) = RegionPos::local_chunk_pos(pos.0.x, pos.0.y);
448        let index = RegionHeader::chunk_index(local_x, local_z);
449
450        let mut regions = self.regions.write().await;
451
452        // Get the region (should already be open via acquire_chunk)
453        let Some(handle) = regions.get_mut(&region_pos) else {
454            log::warn!("load_chunk called without acquire_chunk for region {region_pos:?}");
455            return Ok(None);
456        };
457
458        // Check if chunk exists
459        let entry = handle.header.entries[index];
460        if !entry.exists() {
461            return Ok(None);
462        }
463
464        // Read chunk data from disk
465        let compressed =
466            Self::read_chunk_data(&mut handle.file, entry.sector_offset, entry.size_bytes).await?;
467
468        // Decompress
469        let data = zstd::decode_all(&compressed[..])?;
470
471        // Deserialize
472        let persistent: PersistentChunk = wincode::deserialize(&data)
473            .map_err(|e| io::Error::new(io::ErrorKind::InvalidData, e.to_string()))?;
474
475        // Convert to runtime format (persistent is dropped after this - no duplication!)
476        let status = entry.status;
477        let chunk = Self::persistent_to_chunk(&persistent, pos, status, min_y, height, level);
478
479        Ok(Some((chunk, status)))
480    }

482    /// Acquires a chunk, incrementing the region's reference count.
483    ///
484    /// This opens or creates the region file. Call this before loading or
485    /// generating a chunk, and call `release_chunk` when done with the chunk.
486    ///
487    /// Returns `Ok(true)` if the chunk exists on disk, `Ok(false)` if it doesn't.
488    #[allow(clippy::missing_panics_doc)]
489    pub async fn acquire_chunk(&self, pos: ChunkPos) -> io::Result<bool> {
490        let region_pos = RegionPos::from_chunk(pos.0.x, pos.0.y);
491        let (local_x, local_z) = RegionPos::local_chunk_pos(pos.0.x, pos.0.y);
492        let index = RegionHeader::chunk_index(local_x, local_z);
493
494        let mut regions = self.regions.write().await;
495
496        // Get or open/create the region
497        let handle = if let Some(handle) = regions.get_mut(&region_pos) {
498            handle
499        } else {
500            // open_region creates the file if it doesn't exist
501            let handle = self.open_region(region_pos).await?;
502            regions.insert(region_pos, handle);
503            regions.get_mut(&region_pos).expect("just inserted")
504        };
505
506        // Check if chunk exists
507        let exists = handle.header.entries[index].exists();
508
509        // Increment ref count
510        handle.loaded_chunk_count += 1;
511
512        Ok(exists)
513    }

515    /// Releases a loaded chunk, decrementing the region's reference count.
516    ///
517    /// When all chunks from a region are released, the header is saved (if dirty)
518    /// and the file handle is closed.
519    ///
520    /// This must be called for each chunk returned by `load_chunk`.
521    pub async fn release_chunk(&self, pos: ChunkPos) -> io::Result<()> {
522        let region_pos = RegionPos::from_chunk(pos.0.x, pos.0.y);
523
524        let mut regions = self.regions.write().await;
525
526        let should_close = if let Some(handle) = regions.get_mut(&region_pos) {
527            handle.loaded_chunk_count = handle.loaded_chunk_count.saturating_sub(1);
528            handle.loaded_chunk_count == 0
529        } else {
530            return Ok(());
531        };
532
533        if should_close
534            && let Some(mut handle) = regions.remove(&region_pos)
535            && handle.header_dirty
536        {
537            Self::write_header(&mut handle.file, &handle.header).await?;
538        }
539
540        Ok(())
541    }

543    /// Checks if a chunk exists on disk without loading it.
544    pub async fn chunk_exists(&self, pos: ChunkPos) -> io::Result<bool> {
545        let region_pos = RegionPos::from_chunk(pos.0.x, pos.0.y);
546        let (local_x, local_z) = RegionPos::local_chunk_pos(pos.0.x, pos.0.y);
547        let index = RegionHeader::chunk_index(local_x, local_z);
548
549        let regions = self.regions.write().await;
550
551        // Check cached header first
552        if let Some(handle) = regions.get(&region_pos) {
553            return Ok(handle.header.entries[index].exists());
554        }
555
556        drop(regions);
557
558        // Need to read header from disk
559        let path = self.region_path(region_pos);
560        if !path.exists() {
561            return Ok(false);
562        }
563
564        let mut file = File::open(&path).await?;
565
566        // Skip magic + version
567        file.seek(io::SeekFrom::Start(FILE_HEADER_SIZE as u64))
568            .await?;
569
570        // Read just the one entry we need (8 bytes at index * 8)
571        file.seek(io::SeekFrom::Current((index * 8) as i64)).await?;
572        let mut entry_bytes = [0u8; 8];
573        file.read_exact(&mut entry_bytes).await?;
574
575        let entry = super::format::ChunkEntry::from_bytes(entry_bytes);
576        Ok(entry.exists())
577    }

579    /// Flushes all dirty headers to disk.
580    pub async fn flush_all(&self) -> io::Result<()> {
581        let mut regions = self.regions.write().await;
582
583        for handle in regions.values_mut() {
584            if handle.header_dirty {
585                Self::write_header(&mut handle.file, &handle.header).await?;
586                handle.header_dirty = false;
587            }
588        }
589
590        Ok(())
591    }

593    /// Flushes all dirty headers and closes all region file handles.
594    ///
595    /// This should be called during graceful shutdown after all chunks have been saved.
596    /// It ensures all data is persisted and file handles are properly closed.
597    pub async fn close_all(&self) -> io::Result<()> {
598        let mut regions = self.regions.write().await;
599
600        for (_, mut handle) in regions.drain() {
601            if handle.header_dirty {
602                Self::write_header(&mut handle.file, &handle.header).await?;
603            }
604            // File handle is dropped here, closing the file
605        }
606
607        Ok(())
608    }

610    /// Converts chunk data to persistent format.
611    fn to_persistent(
612        sections: &Sections,
613        block_entities: &[SharedBlockEntity],
614        entities: &[SharedEntity],
615        chunk_pos: ChunkPos,
616    ) -> PersistentChunk {
617        let mut builder = ChunkBuilder::new(&REGISTRY);
618
619        let persistent_sections = sections
620            .sections
621            .iter()
622            .map(|section| Self::section_to_persistent(section, &mut builder))
623            .collect();
624
625        // Serialize block entities
626        let persistent_block_entities: Vec<PersistentBlockEntity> = block_entities
627            .iter()
628            .map(|entity| {
629                let guard = entity.lock();
630                let pos = guard.get_block_pos();
631
632                // Serialize NBT data
633                let mut nbt = NbtCompound::new();
634                guard.save_additional(&mut nbt);
635                let mut nbt_bytes = Vec::new();
636                nbt.write(&mut nbt_bytes);
637
638                PersistentBlockEntity {
639                    x: (pos.0.x - chunk_pos.0.x * 16) as u8,
640                    y: pos.0.y as i16,
641                    z: (pos.0.z - chunk_pos.0.y * 16) as u8,
642                    entity_type: guard.get_type().key.clone(),
643                    nbt_data: nbt_bytes,
644                }
645            })
646            .collect();
647
648        // Serialize entities
649        let persistent_entities: Vec<PersistentEntity> = entities
650            .iter()
651            .filter_map(|entity| {
652                let pos = entity.position();
653                let vel = entity.velocity();
654                let (yaw, pitch) = entity.rotation();
655
656                // Validate position is finite (discard corrupted entities)
657                if !pos.x.is_finite() || !pos.y.is_finite() || !pos.z.is_finite() {
658                    tracing::warn!(
659                        uuid = ?entity.uuid(),
660                        "Entity has non-finite position {:?}, skipping save",
661                        pos
662                    );
663                    return None;
664                }
665
666                // Serialize type-specific NBT data
667                let mut nbt = NbtCompound::new();
668                entity.save_additional(&mut nbt);
669                let mut nbt_bytes = Vec::new();
670                nbt.write(&mut nbt_bytes);
671
672                Some(PersistentEntity {
673                    entity_type: entity.entity_type().key.clone(),
674                    uuid: *entity.uuid().as_bytes(),
675                    pos: [pos.x, pos.y, pos.z],
676                    motion: [vel.x, vel.y, vel.z],
677                    rotation: [yaw, pitch],
678                    on_ground: entity.on_ground(),
679                    nbt_data: nbt_bytes,
680                })
681            })
682            .collect();
683
684        PersistentChunk {
685            last_modified: SystemTime::now()
686                .duration_since(UNIX_EPOCH)
687                .map_or(0, |d| d.as_secs() as u32),
688            block_states: builder.block_states,
689            biomes: builder.biomes,
690            sections: persistent_sections,
691            block_entities: persistent_block_entities,
692            entities: persistent_entities,
693        }
694    }

696    /// Converts a runtime section to persistent format.
697    fn section_to_persistent(
698        section: &SectionHolder,
699        builder: &mut ChunkBuilder,
700    ) -> PersistentSection {
701        let section = section.read();
702        let biomes = Self::biomes_to_persistent(&section.biomes, builder);
703
704        match &section.states {
705            PalettedContainer::Homogeneous(block_id) => {
706                let block_idx = builder.ensure_block_state(*block_id);
707                PersistentSection::Homogeneous {
708                    block_state: block_idx,
709                    biomes,
710                }
711            }
712            PalettedContainer::Heterogeneous(data) => {
713                // Build section-local palette (indices into chunk's block_states)
714                let palette: Vec<u16> = data
715                    .palette
716                    .iter()
717                    .map(|(block_id, _)| builder.ensure_block_state(*block_id))
718                    .collect();
719
720                // Pack block indices (indices into section-local palette)
721                let bits = bits_for_palette_len(palette.len())
722                    .expect("Heterogeneous section should have palette length >= 2");
723                let indices: Vec<u32> = data
724                    .cube
725                    .iter()
726                    .flatten()
727                    .flatten()
728                    .map(|block_id| {
729                        data.palette
730                            .iter()
731                            .position(|(v, _)| v == block_id)
732                            .unwrap_or(0) as u32
733                    })
734                    .collect();
735
736                let block_data = pack_indices(&indices, bits);
737
738                PersistentSection::Heterogeneous {
739                    palette,
740                    bits_per_entry: bits,
741                    block_data,
742                    biomes,
743                }
744            }
745        }
746    }

792    /// Converts a persistent chunk to runtime format.
793    /// The returned chunk is not dirty (freshly loaded from disk).
794    ///
795    /// # Arguments
796    /// * `persistent` - The persistent chunk data
797    /// * `pos` - The chunk position
798    /// * `status` - The chunk status
799    /// * `min_y` - The minimum Y coordinate of the world
800    /// * `height` - The total height of the world
801    /// * `level` - Weak reference to the world for `LevelChunk`
802    fn persistent_to_chunk(
803        persistent: &PersistentChunk,
804        pos: ChunkPos,
805        status: ChunkStatus,
806        min_y: i32,
807        height: i32,
808        level: Weak<World>,
809    ) -> ChunkAccess {
810        let sections: Vec<ChunkSection> = persistent
811            .sections
812            .iter()
813            .map(|section| Self::persistent_to_section(section, persistent))
814            .collect();
815
816        match status {
817            ChunkStatus::Full => {
818                let chunk = LevelChunk::from_disk(
819                    Sections::from_owned(sections.into_boxed_slice()),
820                    pos,
821                    min_y,
822                    height,
823                    level.clone(),
824                );
825
826                // Load block entities
827                for persistent_be in &persistent.block_entities {
828                    if let Some(block_entity) =
829                        Self::persistent_to_block_entity(persistent_be, pos, &chunk)
830                    {
831                        chunk.add_and_register_block_entity(block_entity);
832                    }
833                }
834
835                // Load entities
836                for persistent_entity in &persistent.entities {
837                    if let Some(entity) = Self::persistent_to_entity(persistent_entity, pos, &chunk)
838                    {
839                        chunk.add_and_register_entity(entity);
840                    }
841                }
842
843                // Clear dirty flag since we just loaded (add_and_register marks dirty)
844                chunk.dirty.store(false, Ordering::Release);
845
846                ChunkAccess::Full(chunk)
847            }
848            _ => ChunkAccess::Proto(ProtoChunk::from_disk(
849                Sections::from_owned(sections.into_boxed_slice()),
850                pos,
851                status,
852                min_y,
853                height,
854            )),
855        }
856    }

858    /// Converts a persistent block entity to runtime format.
859    fn persistent_to_block_entity(
860        persistent: &PersistentBlockEntity,
861        chunk_pos: ChunkPos,
862        chunk: &LevelChunk,
863    ) -> Option<SharedBlockEntity> {
864        // Calculate absolute position
865        let abs_x = chunk_pos.0.x * 16 + i32::from(persistent.x);
866        let abs_z = chunk_pos.0.y * 16 + i32::from(persistent.z);
867        let pos = BlockPos::new(abs_x, i32::from(persistent.y), abs_z);
868
869        // Get the block state at this position
870        let state = chunk.get_block_state(pos);
871
872        // Look up the block entity type
873        let block_entity_type = REGISTRY
874            .block_entity_types
875            .by_key(&persistent.entity_type)?;
876
877        // Get the world reference from the chunk
878        let level = chunk.level_weak();
879
880        // Parse and load NBT data
881        if persistent.nbt_data.is_empty() {
882            // No NBT data, just create the entity without loading
883            BLOCK_ENTITIES.create(block_entity_type, level, pos, state)
884        } else {
885            // Parse NBT from bytes as borrowed
886            let Ok(nbt) = read_borrowed_compound(&mut Cursor::new(&persistent.nbt_data)) else {
887                return BLOCK_ENTITIES.create(block_entity_type, level, pos, state);
888            };
889
890            // Create the block entity and load NBT
891            BLOCK_ENTITIES.create_and_load(block_entity_type, level, pos, state, &nbt)
892        }
893    }

895    /// Converts a persistent entity to runtime format.
896    fn persistent_to_entity(
897        persistent: &PersistentEntity,
898        chunk_pos: ChunkPos,
899        chunk: &LevelChunk,
900    ) -> Option<SharedEntity> {
901        use steel_utils::math::Vector3;
902        use uuid::Uuid;
903
904        // Reconstruct base fields
905        let pos = Vector3::new(persistent.pos[0], persistent.pos[1], persistent.pos[2]);
906        let mut velocity = Vector3::new(
907            persistent.motion[0],
908            persistent.motion[1],
909            persistent.motion[2],
910        );
911        let rotation = (persistent.rotation[0], persistent.rotation[1]);
912        let uuid = Uuid::from_bytes(persistent.uuid);
913
914        // Validate position is finite
915        if !pos.x.is_finite() || !pos.y.is_finite() || !pos.z.is_finite() {
916            tracing::warn!(
917                ?uuid,
918                "Entity has non-finite position {:?}, skipping load",
919                pos
920            );
921            return None;
922        }
923
924        // Validate position is within expected chunk (sanity check)
925        let expected_chunk_x = (pos.x as i32) >> 4;
926        let expected_chunk_z = (pos.z as i32) >> 4;
927        if chunk_pos.0.x != expected_chunk_x || chunk_pos.0.y != expected_chunk_z {
928            tracing::warn!(
929                ?uuid,
930                "Entity position {:?} doesn't match chunk {:?}, loading anyway",
931                pos,
932                chunk_pos
933            );
934        }
935
936        // Clamp motion values > 10.0 to 0 (vanilla behavior to prevent corruption)
937        if velocity.x.abs() > 10.0 {
938            velocity.x = 0.0;
939        }
940        if velocity.y.abs() > 10.0 {
941            velocity.y = 0.0;
942        }
943        if velocity.z.abs() > 10.0 {
944            velocity.z = 0.0;
945        }
946
947        // Look up entity type
948        let entity_type = REGISTRY.entity_types.by_key(&persistent.entity_type)?;
949
950        // Check if we have a load factory for this entity type
951        if !ENTITIES.has_load_factory(entity_type) {
952            tracing::debug!(
953                entity_type = %persistent.entity_type,
954                "No load factory for entity type, skipping"
955            );
956            return None;
957        }
958
959        // Get world reference
960        let level = chunk.level_weak();
961
962        // Parse NBT from bytes (or use empty compound data)
963        let nbt_bytes = if persistent.nbt_data.is_empty() {
964            // Empty NBT compound: type byte (10 = compound), empty name (2 zero bytes), end tag (0)
965            &[0x0a, 0x00, 0x00, 0x00][..]
966        } else {
967            &persistent.nbt_data[..]
968        };
969
970        let Ok(nbt) = read_borrowed_compound(&mut Cursor::new(nbt_bytes)) else {
971            tracing::warn!(?uuid, "Failed to parse entity NBT, skipping");
972            return None;
973        };
974
975        ENTITIES.create_and_load(
976            entity_type,
977            pos,
978            uuid,
979            velocity,
980            rotation,
981            persistent.on_ground,
982            level,
983            &nbt,
984        )
985    }

987    /// Converts a persistent section to runtime format.
988    fn persistent_to_section(
989        persistent: &PersistentSection,
990        chunk: &PersistentChunk,
991    ) -> ChunkSection {
992        match persistent {
993            PersistentSection::Homogeneous {
994                block_state,
995                biomes,
996            } => {
997                let block_id = Self::resolve_block_state(chunk, *block_state);
998                let biome_data = Self::persistent_to_biomes(biomes, chunk);
999
1000                ChunkSection::new_with_biomes(PalettedContainer::Homogeneous(block_id), biome_data)
1001            }
1002            PersistentSection::Heterogeneous {
1003                palette,
1004                bits_per_entry,
1005                block_data,
1006                biomes,
1007            } => {
1008                // Unpack indices (into section-local palette)
1009                let indices = unpack_indices(block_data, *bits_per_entry, BLOCKS_PER_SECTION);
1010
1011                // Build runtime palette by resolving section-local -> chunk -> runtime
1012                let runtime_palette: Vec<BlockStateId> = palette
1013                    .iter()
1014                    .map(|&idx| Self::resolve_block_state(chunk, idx))
1015                    .collect();
1016
1017                // Build cube
1018                let mut cube = Box::new([[[BlockStateId(0); 16]; 16]; 16]);
1019                for (i, &idx) in indices.iter().enumerate() {
1020                    let y = i / 256;
1021                    let z = (i / 16) % 16;
1022                    let x = i % 16;
1023                    cube[y][z][x] = runtime_palette
1024                        .get(idx as usize)
1025                        .copied()
1026                        .unwrap_or(BlockStateId(0));
1027                }
1028
1029                let states = PalettedContainer::from_cube(cube);
1030                let biome_data = Self::persistent_to_biomes(biomes, chunk);
1031
1032                ChunkSection::new_with_biomes(states, biome_data)
1033            }
1034        }
1035    }

1037    /// Converts persistent biome data to runtime format.
1038    fn persistent_to_biomes(
1039        persistent: &PersistentBiomeData,
1040        chunk: &PersistentChunk,
1041    ) -> PalettedContainer<u8, 4> {
1042        match persistent {
1043            PersistentBiomeData::Homogeneous { biome } => {
1044                let biome_id = Self::resolve_biome(chunk, *biome);
1045                PalettedContainer::Homogeneous(biome_id)
1046            }
1047            PersistentBiomeData::Heterogeneous {
1048                palette,
1049                bits_per_entry,
1050                biome_data,
1051            } => {
1052                let indices = unpack_indices(biome_data, *bits_per_entry, BIOMES_PER_SECTION);
1053
1054                // Resolve section-local palette -> chunk palette -> runtime
1055                let runtime_palette: Vec<u8> = palette
1056                    .iter()
1057                    .map(|&idx| Self::resolve_biome(chunk, idx))
1058                    .collect();
1059
1060                let mut cube = Box::new([[[0u8; 4]; 4]; 4]);
1061                for (i, &idx) in indices.iter().enumerate() {
1062                    let y = i / 16;
1063                    let z = (i / 4) % 4;
1064                    let x = i % 4;
1065                    cube[y][z][x] = runtime_palette.get(idx as usize).copied().unwrap_or(0);
1066                }
1067
1068                PalettedContainer::from_cube(cube)
1069            }
1070        }
1071    }

1073    /// Resolves a chunk palette index to a runtime `BlockStateId`.
1074    fn resolve_block_state(chunk: &PersistentChunk, index: u16) -> BlockStateId {
1075        if let Some(state) = chunk.block_states.get(index as usize) {
1076            // Convert properties to the format expected by the registry
1077            let properties: Vec<(&str, &str)> = state
1078                .properties
1079                .iter()
1080                .map(|(k, v)| (k.as_str(), v.as_str()))
1081                .collect();
1082
1083            if let Some(state_id) = REGISTRY
1084                .blocks
1085                .state_id_from_properties(&state.name, &properties)
1086            {
1087                return state_id;
1088            }
1089        }
1090        BlockStateId(0) // Air fallback
1091    }

1093    /// Resolves a chunk palette index to a runtime biome ID.
1094    fn resolve_biome(chunk: &PersistentChunk, index: u16) -> u8 {
1095        if let Some(biome_key) = chunk.biomes.get(index as usize)
1096            && let Some(id) = REGISTRY.biomes.id_from_key(biome_key)
1097        {
1098            return id as u8;
1099        }
1100        0 // Plains fallback
