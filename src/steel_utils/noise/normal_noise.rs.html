1//! Normal (Double Perlin) noise implementation matching vanilla Minecraft's NormalNoise.java
2//!
3//! This combines two `PerlinNoise` samplers with slightly different coordinate scaling
4//! to create smoother, more natural-looking noise. It's used for biome climate parameters.
5
6use crate::noise::PerlinNoise;
7use crate::random::{PositionalRandom, RandomSource, RandomSplitter};
8
9/// Input factor for the second Perlin sampler.
10///
11/// This is the exact value from vanilla `NormalNoise.java`.
12/// The second sampler's coordinates are multiplied by this factor to create
13/// variation between the two samplers.
14#[allow(clippy::unreadable_literal)]
15pub const INPUT_FACTOR: f64 = 1.0181268882175227;
16
17/// Value factor numerator matching vanilla's inline literal `0.16666666666666666` (1/6).
18///
19/// Vanilla declares a constant `TARGET_DEVIATION = 0.3333333333333333` (1/3) but never
20/// uses it â€” the constructor hardcodes `0.16666666666666666` (1/6) as the numerator in
21/// `valueFactor = 0.16666... / expectedDeviation(span)`. We name this differently to
22/// avoid confusion with vanilla's dead `TARGET_DEVIATION` constant.
23#[allow(clippy::unreadable_literal)]
24const VALUE_FACTOR_NUMERATOR: f64 = 0.16666666666666666;
25
26/// Normal (Double Perlin) noise generator.
27///
28/// Combines two `PerlinNoise` samplers with different coordinate scales to create
29/// smoother noise. The result is scaled by a value factor based on the octave span.
30#[derive(Debug, Clone)]
31pub struct NormalNoise {
32    /// First Perlin noise sampler
33    first: PerlinNoise,
34    /// Second Perlin noise sampler (coordinates scaled by `INPUT_FACTOR`)
35    second: PerlinNoise,
36    /// Factor applied to the sum of both samplers
37    value_factor: f64,
38    /// Maximum possible output value
39    max_value: f64,
40}
41
42impl NormalNoise {
43    /// Create a new `NormalNoise` from a mutable sequential random source.
44    ///
45    /// This matches vanilla's `NormalNoise` constructor:
46    /// 1. Create first `PerlinNoise` (which advances the random state by consuming 262 + forking)
47    /// 2. Create second `PerlinNoise` (which sees the advanced state)
48    ///
49    /// This ensures the two `PerlinNoise` instances have different seeds.
50    #[must_use]
51    pub fn create_from_random(
52        random: &mut RandomSource,
53        first_octave: i32,
54        amplitudes: &[f64],
55    ) -> Self {
56        let first = PerlinNoise::create_from_random(random, first_octave, amplitudes);
57        let second = PerlinNoise::create_from_random(random, first_octave, amplitudes);
58
59        Self::finish(first, second, amplitudes)
60    }
61
62    /// Create a new `NormalNoise` from a positional random splitter.
63    ///
64    /// **Note**: This creates a sequential random source from the splitter's noise ID,
65    /// then delegates to `create_from_random` for vanilla-matching behavior.
66    #[must_use]
67    pub fn create(
68        splitter: &RandomSplitter,
69        noise_id: &str,
70        first_octave: i32,
71        amplitudes: &[f64],
72    ) -> Self {
73        let mut random = splitter.with_hash_of(noise_id);
74        Self::create_from_random(&mut random, first_octave, amplitudes)
75    }
76
77    /// Create a `NormalNoise` using the legacy nether biome initialization path.
78    ///
79    /// This uses `PerlinNoise::create_legacy_for_nether` instead of the hash-based
80    /// positional seeding. The `ImprovedNoise` instances are created directly from
81    /// a sequential `LegacyRandomSource`. Matches vanilla's
82    /// `NormalNoise.createLegacyNetherBiome()`.
83    #[must_use]
84    pub fn create_legacy_nether_biome(
85        random: &mut RandomSource,
86        first_octave: i32,
87        amplitudes: &[f64],
88    ) -> Self {
89        let first = PerlinNoise::create_legacy_for_nether(random, first_octave, amplitudes);
90        let second = PerlinNoise::create_legacy_for_nether(random, first_octave, amplitudes);
91
92        Self::finish(first, second, amplitudes)
93    }
94
95    /// Finish construction with the two `PerlinNoise` instances.
96    fn finish(first: PerlinNoise, second: PerlinNoise, amplitudes: &[f64]) -> Self {
97        // Find the span of non-zero octaves
98        let mut min_octave = i32::MAX;
99        let mut max_octave = i32::MIN;
100        for (i, &amp) in amplitudes.iter().enumerate() {
101            if amp != 0.0 {
102                min_octave = min_octave.min(i as i32);
103                max_octave = max_octave.max(i as i32);
104            }
105        }
106
107        // All-zero amplitudes: silent noise, always returns 0.
108        if min_octave == i32::MAX {
109            return Self {
110                first,
111                second,
112                value_factor: 0.0,
113                max_value: 0.0,
114            };
115        }
116
117        // Calculate value factor based on octave span
118        let octave_span = max_octave - min_octave;
119        let value_factor = VALUE_FACTOR_NUMERATOR / expected_deviation(octave_span);
120        let max_value = (first.max_value() + second.max_value()) * value_factor;
121
122        Self {
123            first,
124            second,
125            value_factor,
126            max_value,
127        }
128    }
129
130    /// Sample the noise at the given coordinates.
131    ///
132    /// The result combines two Perlin noise samples:
133    /// - First sampler at (x, y, z)
134    /// - Second sampler at (x * `INPUT_FACTOR`, y * `INPUT_FACTOR`, z * `INPUT_FACTOR`)
135    ///
136    /// The sum is then scaled by the value factor.
137    #[inline]
138    #[must_use]
139    pub fn get_value(&self, x: f64, y: f64, z: f64) -> f64 {
140        let x2 = x * INPUT_FACTOR;
141        let y2 = y * INPUT_FACTOR;
142        let z2 = z * INPUT_FACTOR;
143        (self.first.get_value(x, y, z) + self.second.get_value(x2, y2, z2)) * self.value_factor
144    }
145
146    /// Get the maximum possible output value.
147    #[inline]
148    #[must_use]
149    pub const fn max_value(&self) -> f64 {
150        self.max_value
151    }
152}
153
154/// Calculate the expected deviation for a given octave span.
155///
156/// This is used to normalize the output of the combined noise.
157/// Formula: 0.1 * (1 + 1/(span + 1))
158#[inline]
159fn expected_deviation(octave_span: i32) -> f64 {
160    0.1 * (1.0 + 1.0 / f64::from(octave_span + 1))
161}

163#[cfg(test)]
164#[allow(clippy::unreadable_literal)]
165mod tests {
166    use super::*;
167    use crate::random::{Random, xoroshiro::Xoroshiro};
168
169    #[test]
170    fn test_normal_noise_deterministic() {
171        let mut rng = Xoroshiro::from_seed(12345);
172        let splitter = rng.next_positional();
173
174        let amplitudes = [1.0, 1.0, 1.0];
175        let noise1 = NormalNoise::create(&splitter, "test_noise", -3, &amplitudes);
176        let noise2 = NormalNoise::create(&splitter, "test_noise", -3, &amplitudes);
177
178        let v1 = noise1.get_value(100.0, 64.0, 100.0);
179        let v2 = noise2.get_value(100.0, 64.0, 100.0);
180        assert!((v1 - v2).abs() < 1e-15);
181    }
182
183    #[test]
184    fn test_normal_noise_spatial_variation() {
185        let mut rng = Xoroshiro::from_seed(42);
186        let splitter = rng.next_positional();
187
188        let noise = NormalNoise::create(&splitter, "test_noise", -4, &[1.0, 1.0, 1.0, 1.0]);
189
190        // Sample at different locations
191        let values: Vec<f64> = (0..10)
192            .map(|i| noise.get_value(f64::from(i) * 50.0, 64.0, f64::from(i) * 50.0))
193            .collect();
194
195        // Check there's variation
196        let min = values.iter().copied().fold(f64::INFINITY, f64::min);
197        let max = values.iter().copied().fold(f64::NEG_INFINITY, f64::max);
198        assert!(max - min > 0.01, "Noise should have spatial variation");
199    }
200
201    #[test]
202    fn test_first_and_second_differ() {
203        let mut rng = Xoroshiro::from_seed(12345);
204        let splitter = rng.next_positional();
205
206        let noise = NormalNoise::create(&splitter, "test_noise", -3, &[1.0, 1.0, 1.0]);
207
208        // The first and second samplers should produce different raw values
209        // (but we can only test via the combined output)
210        let v1 = noise.get_value(1000.0, 0.0, 1000.0);
211        let v2 = noise.get_value(1001.0, 0.0, 1000.0);
212        // Values at different coordinates should differ
213        assert!((v1 - v2).abs() > 0.0001);
214    }
215
216    #[test]
217    fn test_expected_deviation() {
218        // Check the formula produces expected values
219        assert!((expected_deviation(0) - 0.2).abs() < 1e-10);
220        assert!((expected_deviation(1) - 0.15).abs() < 1e-10);
221        assert!((expected_deviation(2) - 0.13333333333333333).abs() < 1e-10);
222    }

224    #[test]
225    fn test_input_factor() {
226        // Verify the constant matches vanilla
227        assert!((INPUT_FACTOR - 1.0181268882175227).abs() < 1e-15);
228    }
