1//! Density function transpiler: compiles `DensityFunction` trees into native Rust functions.
2//!
3//! This module takes a registry of named `DensityFunction` trees and noise router entry
4//! points, and generates Rust source code (`proc_macro2::TokenStream`) that evaluates
5//! them as compiled functions — eliminating runtime tree interpretation, HashMap-based
6//! caching, and Arc pointer chasing.
7//!
8//! # Usage
9//!
10//! ```ignore
11//! let input = TranspilerInput {
12//!     registry,       // BTreeMap<String, DensityFunction>
13//!     router_entries, // BTreeMap<String, DensityFunction>
14//! };
15//! let tokens: TokenStream = transpile(&input);
16//! ```
17//!
18//! Gated behind the `codegen` feature flag.
19

32/// Input to the transpiler.
33pub struct TranspilerInput {
34    /// Named density functions (registry entries like `"minecraft:overworld/continents"`).
35    pub registry: BTreeMap<String, DensityFunction>,
36    /// Noise router entry points (like `"temperature"`, `"final_density"`).
37    pub router_entries: BTreeMap<String, DensityFunction>,
38    /// Prefix for generated struct names (e.g., `"Overworld"` → `OverworldNoises`, `OverworldColumnCache`).
39    pub prefix: String,
40}
41
42/// Compile density function trees into a `TokenStream` of Rust code.
43///
44/// The generated code contains:
45/// - `{Prefix}Noises` struct with one `NormalNoise` field per noise used
46/// - `{Prefix}ColumnCache` struct with fields for flat-cached (xz-only) values
47/// - Private `compute_*` functions for each named density function
48/// - Public `router_*` functions for each noise router entry point
49#[must_use]
50pub fn transpile(input: &TranspilerInput) -> TokenStream {
51    let mut ctx = TranspileContext::new(&input.prefix);
52
53    // Phase 1: Analyze the graph
54    ctx.analyze(input);
55
56    // Phase 2: Generate code
57    let noises_struct = ctx.gen_noises_struct();
58    let noises_impl = ctx.gen_noises_impl();
59    let named_fns = ctx.gen_named_functions(input);
60    let column_cache = ctx.gen_column_cache(input);
61    let router_fns = ctx.gen_router_functions(input);
62
63    // Imports are emitted here so each dimension's output is self-contained
64    // when wrapped in a module by the caller.
65    quote! {
66        use steel_utils::density::spline_eval;
67        use steel_utils::density::RarityValueMapper;
68        use steel_utils::math::{clamp, map_clamped};
69        use steel_utils::noise::NormalNoise;
70        use steel_utils::random::RandomSplitter;
71
72        #noises_struct
73        #noises_impl
74        #column_cache
75        #named_fns
76        #router_fns
77    }
78}

82/// Tracks state during transpilation.
83struct TranspileContext {
84    /// All noise IDs referenced by any density function.
85    noise_ids: BTreeSet<String>,
86    /// Named functions that are flat-cached (xz-only).
87    flat_cached: BTreeSet<String>,
88    /// Router entries that are Y-independent (inferred flat).
89    /// Their results are cached in the column cache.
90    flat_routers: BTreeSet<String>,
91    /// Named functions in topological order (dependencies first).
92    topo_order: Vec<String>,
93    /// Named functions that are actually reachable from router entries.
94    used_names: BTreeSet<String>,
95    /// Counter for generating unique spline function names.
96    spline_counter: usize,
97    /// Generated spline helper functions.
98    spline_fns: Vec<TokenStream>,
99    /// Generated ident for the noises struct (e.g., `OverworldNoises`).
100    noises_ident: Ident,
101    /// Generated ident for the column cache struct (e.g., `OverworldColumnCache`).
102    cache_ident: Ident,
103}
104
105impl TranspileContext {
106    fn new(prefix: &str) -> Self {
107        Self {
108            noise_ids: BTreeSet::new(),
109            flat_cached: BTreeSet::new(),
110            flat_routers: BTreeSet::new(),
111            topo_order: Vec::new(),
112            used_names: BTreeSet::new(),
113            spline_counter: 0,
114            spline_fns: Vec::new(),
115            noises_ident: format_ident!("{prefix}Noises"),
116            cache_ident: format_ident!("{prefix}ColumnCache"),
117        }
118    }

120    // ── Phase 1: Analysis ───────────────────────────────────────────────
121
122    fn analyze(&mut self, input: &TranspilerInput) {
123        for df in input.router_entries.values() {
124            self.walk_df(df, input);
125        }
126
127        // Mark explicitly flat-cached functions
128        for name in &self.used_names {
129            if let Some(df) = input.registry.get(name)
130                && is_flat_cached(df)
131            {
132                self.flat_cached.insert(name.clone());
133            }
134        }
135
136        // Infer flatness: a function is flat if it doesn't use y and all its
137        // Reference dependencies are also flat. Iterate until convergence.
138        loop {
139            let mut changed = false;
140            for name in &self.used_names.clone() {
141                if self.flat_cached.contains(name) {
142                    continue;
143                }
144                let Some(df) = input.registry.get(name) else {
145                    continue;
146                };
147                let inner = unwrap_markers(df);
148                if !uses_y(inner)
149                    && collect_references(inner)
150                        .iter()
151                        .all(|dep| self.flat_cached.contains(dep))
152                {
153                    self.flat_cached.insert(name.clone());
154                    changed = true;
155                }
156            }
157            if !changed {
158                break;
159            }
160        }
161
162        // Infer flatness for router entries: a router entry is flat if it doesn't
163        // use y and all its Reference dependencies are flat-cached named functions.
164        // This catches cases like temperature/vegetation that are Y-independent but
165        // lack explicit FlatCache markers in vanilla's JSON.
166        for (name, df) in &input.router_entries {
167            let inner = unwrap_markers(df);
168            if is_flat_cached(df)
169                || (!uses_y(inner)
170                    && collect_references(inner)
171                        .iter()
172                        .all(|dep| self.flat_cached.contains(dep)))
173            {
174                self.flat_routers.insert(name.clone());
175            }
176        }
177
178        self.topo_order = self.topological_sort(input);
179    }
180
181    fn walk_df(&mut self, df: &DensityFunction, input: &TranspilerInput) {
182        match df {
183            DensityFunction::Constant(_)
184            | DensityFunction::BlendAlpha(_)
185            | DensityFunction::BlendOffset(_)
186            | DensityFunction::EndIslands
187            | DensityFunction::YClampedGradient(_) => {}
188
189            DensityFunction::Noise(n) => {
190                self.noise_ids.insert(n.noise_id.clone());
191            }
192            DensityFunction::ShiftedNoise(sn) => {
193                self.walk_df(&sn.shift_x, input);
194                self.walk_df(&sn.shift_y, input);
195                self.walk_df(&sn.shift_z, input);
196                self.noise_ids.insert(sn.noise_id.clone());
197            }
198            DensityFunction::ShiftA(s) => {
199                self.noise_ids.insert(s.noise_id.clone());
200            }
201            DensityFunction::ShiftB(s) => {
202                self.noise_ids.insert(s.noise_id.clone());
203            }
204            DensityFunction::Shift(s) => {
205                self.noise_ids.insert(s.noise_id.clone());
206            }
207            DensityFunction::TwoArgumentSimple(t) => {
208                self.walk_df(&t.argument1, input);
209                self.walk_df(&t.argument2, input);
210            }
211            DensityFunction::Mapped(m) => self.walk_df(&m.input, input),
212            DensityFunction::Clamp(c) => self.walk_df(&c.input, input),
213            DensityFunction::RangeChoice(rc) => {
214                self.walk_df(&rc.input, input);
215                self.walk_df(&rc.when_in_range, input);
216                self.walk_df(&rc.when_out_of_range, input);
217            }
218            DensityFunction::Spline(s) => self.walk_spline(&s.spline, input),
219            DensityFunction::BlendedNoise(_) => {
220                self.noise_ids.insert("minecraft:offset".to_string());
221            }
222            DensityFunction::WeirdScaledSampler(ws) => {
223                self.walk_df(&ws.input, input);
224                self.noise_ids.insert(ws.noise_id.clone());
225            }
226            DensityFunction::BlendDensity(bd) => self.walk_df(&bd.input, input),
227            DensityFunction::Marker(m) => self.walk_df(&m.wrapped, input),
228            DensityFunction::Reference(r) => {
229                if !self.used_names.contains(&r.id) {
230                    self.used_names.insert(r.id.clone());
231                    if let Some(ref_df) = input.registry.get(&r.id) {
232                        self.walk_df(ref_df, input);
233                    }
234                }
235            }
236        }
237    }
238
239    fn walk_spline(&mut self, spline: &CubicSpline, input: &TranspilerInput) {
240        self.walk_df(&spline.coordinate, input);
241        for point in &spline.points {
242            if let SplineValue::Spline(nested) = &point.value {
243                self.walk_spline(nested, input);
244            }
245        }
246    }
247
248    fn topological_sort(&self, input: &TranspilerInput) -> Vec<String> {
249        let mut visited = BTreeSet::new();
250        let mut order = Vec::new();
251        for name in &self.used_names {
252            self.topo_visit(name, input, &mut visited, &mut order);
253        }
254        order
255    }
256
257    fn topo_visit(
258        &self,
259        name: &str,
260        input: &TranspilerInput,
261        visited: &mut BTreeSet<String>,
262        order: &mut Vec<String>,
263    ) {
264        if visited.contains(name) {
265            return;
266        }
267        visited.insert(name.to_string());
268        if let Some(df) = input.registry.get(name) {
269            for dep in collect_references(df) {
270                if self.used_names.contains(&dep) {
271                    self.topo_visit(&dep, input, visited, order);
272                }
273            }
274        }
275        order.push(name.to_string());
276    }

280    fn gen_noises_struct(&self) -> TokenStream {
281        let fields: Vec<TokenStream> = self
282            .noise_ids
283            .iter()
284            .map(|id| {
285                let field = noise_field_ident(id);
286                quote! { pub #field: NormalNoise }
287            })
288            .collect();
289
290        let noises = &self.noises_ident;
291        quote! {
292            /// All noise generators needed by this dimension's density functions.
293            ///
294            /// Created at runtime from a seed via the `create` method.
295            pub struct #noises {
296                #(#fields),*
297            }
298        }
299    }
300
301    fn gen_noises_impl(&self) -> TokenStream {
302        let field_inits: Vec<TokenStream> = self
303            .noise_ids
304            .iter()
305            .map(|id| {
306                let field = noise_field_ident(id);
307                let id_lit = Literal::string(id);
308                quote! {
309                    #field: {
310                        let p = params.get(#id_lit).expect(concat!("missing noise params: ", #id_lit));
311                        NormalNoise::create(splitter, #id_lit, p.first_octave, &p.amplitudes)
312                    }
313                }
314            })
315            .collect();
316
317        let noises = &self.noises_ident;
318        quote! {
319            impl #noises {
320                /// Create all noise generators from a seed's positional splitter and noise parameters.
321                pub fn create(
322                    splitter: &RandomSplitter,
323                    params: &rustc_hash::FxHashMap<String, steel_utils::density::NoiseParameters>,
324                ) -> Self {
325                    Self {
326                        #(#field_inits),*
327                    }
328                }
329            }
330        }
331    }

333    /// Generate the column cache struct and its `ensure` method.
334    fn gen_column_cache(&mut self, _input: &TranspilerInput) -> TokenStream {
335        let flat_names: Vec<&String> = self
336            .topo_order
337            .iter()
338            .filter(|n| self.flat_cached.contains(*n))
339            .collect();
340
341        let cache_fields: Vec<TokenStream> = flat_names
342            .iter()
343            .map(|name| {
344                let field = named_fn_field_ident(name);
345                quote! { pub #field: f64 }
346            })
347            .collect();
348
349        // Generate the ensure body: compute each flat-cached value in topo order.
350        // We reborrow &*self temporarily for function calls, then write to self.field.
351        let ensure_stmts: Vec<TokenStream> = flat_names
352            .iter()
353            .map(|name| {
354                let field = named_fn_field_ident(name);
355                let compute_fn = named_fn_ident(name);
356                quote! {
357                    let val = #compute_fn(noises, &*self, x, z);
358                    self.#field = val;
359                }
360            })
361            .collect();
362
363        let default_fields: Vec<TokenStream> = flat_names
364            .iter()
365            .map(|name| {
366                let field = named_fn_field_ident(name);
367                quote! { #field: 0.0 }
368            })
369            .collect();
370
371        // Flat router entries: cached alongside named functions.
372        // Computed after all named functions since they may depend on them.
373        let router_fields: Vec<TokenStream> = self
374            .flat_routers
375            .iter()
376            .map(|name| {
377                let field = router_cache_field_ident(name);
378                quote! { pub #field: f64 }
379            })
380            .collect();
381
382        let router_ensure_stmts: Vec<TokenStream> = self
383            .flat_routers
384            .iter()
385            .map(|name| {
386                let field = router_cache_field_ident(name);
387                let compute_fn = router_compute_fn_ident(name);
388                quote! {
389                    let val = #compute_fn(noises, &*self, x, z);
390                    self.#field = val;
391                }
392            })
393            .collect();
394
395        let router_default_fields: Vec<TokenStream> = self
396            .flat_routers
397            .iter()
398            .map(|name| {
399                let field = router_cache_field_ident(name);
400                quote! { #field: 0.0 }
401            })
402            .collect();
403
404        let noises = &self.noises_ident;
405        let cache = &self.cache_ident;
406        quote! {
407            /// Column-level cache for flat-cached (xz-only) density function results.
408            ///
409            /// Call [`ensure`](Self::ensure) before reading values. Values are recomputed
410            /// only when `(x, z)` changes.
411            pub struct #cache {
412                /// Cached x block coordinate.
413                pub x: i32,
414                /// Cached z block coordinate.
415                pub z: i32,
416                valid: bool,
417                #(#cache_fields,)*
418                #(#router_fields),*
419            }
420
421            impl #cache {
422                /// Create a new, empty column cache.
423                #[must_use]
424                pub fn new() -> Self {
425                    Self {
426                        x: 0,
427                        z: 0,
428                        valid: false,
429                        #(#default_fields,)*
430                        #(#router_default_fields),*
431                    }
432                }
433
434                /// Ensure the cache is populated for the given `(x, z)` block coordinates.
435                ///
436                /// If the cache already holds values for this column, this is a no-op.
437                /// Computes all flat-cached density functions in topological order,
438                /// then any Y-independent router entries.
439                pub fn ensure(&mut self, x: i32, z: i32, noises: &#noises) {
440                    if self.valid && self.x == x && self.z == z {
441                        return;
442                    }
443                    self.x = x;
444                    self.z = z;
445                    #(#ensure_stmts)*
446                    #(#router_ensure_stmts)*
447                    self.valid = true;
448                }
449            }
450        }
451    }

453    /// Generate the function parameter list for a density function.
454    fn fn_params(&self, is_flat: bool) -> TokenStream {
455        let noises = &self.noises_ident;
456        let cache = &self.cache_ident;
457        if is_flat {
458            quote! { noises: &#noises, cache: &#cache, x: i32, z: i32 }
459        } else {
460            quote! { noises: &#noises, cache: &#cache, x: i32, y: i32, z: i32 }
461        }
462    }

464    /// Generate the function parameter list for a router entry point.
465    /// Router functions read x/z from the cache, so flat variants omit explicit coords.
466    fn fn_params_router(&self, is_flat: bool) -> TokenStream {
467        let noises = &self.noises_ident;
468        let cache = &self.cache_ident;
469        if is_flat {
470            quote! { noises: &#noises, cache: &#cache }
471        } else {
472            quote! { noises: &#noises, cache: &#cache, x: i32, y: i32, z: i32 }
473        }
474    }

476    /// Generate all named density functions.
477    fn gen_named_functions(&mut self, input: &TranspilerInput) -> TokenStream {
478        let mut fns = Vec::new();
479
480        for name in self.topo_order.clone() {
481            let Some(df) = input.registry.get(&name) else {
482                continue;
483            };
484            let inner = unwrap_markers(df).clone();
485            let fn_name = named_fn_ident(&name);
486            let is_flat = self.flat_cached.contains(&name);
487
488            let body = self.gen_expr(&inner, input, is_flat);
489
490            let params = self.fn_params(is_flat);
491
492            let doc = Literal::string(&format!("`{name}`"));
493            fns.push(quote! {
494                #[doc = #doc]
495                #[inline]
496                fn #fn_name(#params) -> f64 {
497                    #body
498                }
499            });
500        }
501
502        let spline_fns = mem::take(&mut self.spline_fns);
503
504        quote! {
505            #(#fns)*
506            #(#spline_fns)*
507        }
508    }

510    /// Generate the router entry point functions.
511    ///
512    /// Flat (Y-independent) routers read their result from the column cache.
513    /// A private `compute_router_*` function is also emitted for each flat
514    /// router, called by `ensure()` to populate the cache.
515    fn gen_router_functions(&mut self, input: &TranspilerInput) -> TokenStream {
516        let mut fns = Vec::new();
517
518        for (name, df) in &input.router_entries {
519            let fn_name = format_ident!("router_{}", sanitize_name(name));
520            let doc = Literal::string(&format!("Noise router entry: `{name}`"));
521
522            if self.flat_routers.contains(name) {
523                // Flat router: result is cached in the column cache.
524                // Generate private compute function used by ensure().
525                let compute_fn_name = router_compute_fn_ident(name);
526                let inner = unwrap_markers(df);
527                let compute_body = self.gen_expr(inner, input, true);
528                let compute_params = self.fn_params(true);
529
530                fns.push(quote! {
531                    #[inline]
532                    fn #compute_fn_name(#compute_params) -> f64 {
533                        #compute_body
534                    }
535                });
536
537                // Public router function returns the cached value.
538                // Keeps the full (noises, cache, x, y, z) signature for API consistency.
539                let cache_field = router_cache_field_ident(name);
540                let full_params = self.fn_params_router(false);
541                fns.push(quote! {
542                    #[doc = #doc]
543                    #[inline]
544                    pub fn #fn_name(#full_params) -> f64 {
545                        cache.#cache_field
546                    }
547                });
548            } else {
549                let inner = unwrap_markers(df);
550                let is_flat = is_flat_cached(df);
551                let body = self.gen_expr(inner, input, is_flat);
552                let params = self.fn_params_router(is_flat);
553
554                fns.push(quote! {
555                    #[doc = #doc]
556                    #[inline]
557                    pub fn #fn_name(#params) -> f64 {
558                        let x = cache.x;
559                        let z = cache.z;
560                        #body
561                    }
562                });
563            }
564        }
565
566        let spline_fns = mem::take(&mut self.spline_fns);
567        quote! {
568            #(#fns)*
569            #(#spline_fns)*
570        }
571    }

575    /// Generate a `TokenStream` expression that computes a density function value.
576    ///
577    /// `is_flat` indicates this expression tree is xz-only (no y available).
578    #[allow(clippy::too_many_lines)]
579    fn gen_expr(
580        &mut self,
581        df: &DensityFunction,
582        input: &TranspilerInput,
583        is_flat: bool,
584    ) -> TokenStream {
585        match df {
586            DensityFunction::Constant(c) => {
587                let val = Literal::f64_unsuffixed(c.value);
588                quote! { #val }
589            }
590
591            DensityFunction::YClampedGradient(g) => {
592                let from_y = Literal::f64_unsuffixed(f64::from(g.from_y));
593                let to_y = Literal::f64_unsuffixed(f64::from(g.to_y));
594                let from_val = Literal::f64_unsuffixed(g.from_value);
595                let to_val = Literal::f64_unsuffixed(g.to_value);
596                quote! { map_clamped(f64::from(y), #from_y, #to_y, #from_val, #to_val) }
597            }
598
599            DensityFunction::Noise(n) => {
600                let field = noise_field_ident(&n.noise_id);
601                let xz_scale = Literal::f64_unsuffixed(n.xz_scale);
602                let y_scale = Literal::f64_unsuffixed(n.y_scale);
603                if is_flat || n.y_scale == 0.0 {
604                    quote! { noises.#field.get_value(f64::from(x) * #xz_scale, 0.0, f64::from(z) * #xz_scale) }
605                } else {
606                    quote! { noises.#field.get_value(f64::from(x) * #xz_scale, f64::from(y) * #y_scale, f64::from(z) * #xz_scale) }
607                }
608            }
609
610            DensityFunction::ShiftedNoise(sn) => {
611                let dx = self.gen_expr(&sn.shift_x, input, is_flat);
612                let dy = self.gen_expr(&sn.shift_y, input, is_flat);
613                let dz = self.gen_expr(&sn.shift_z, input, is_flat);
614                let field = noise_field_ident(&sn.noise_id);
615                let xz_scale = Literal::f64_unsuffixed(sn.xz_scale);
616                let y_scale = Literal::f64_unsuffixed(sn.y_scale);
617                // Vanilla formula: x * xz_scale + dx (multiply THEN add shift)
618                if is_flat || sn.y_scale == 0.0 {
619                    quote! {{
620                        let dx = #dx;
621                        let dz = #dz;
622                        noises.#field.get_value(
623                            f64::from(x) * #xz_scale + dx,
624                            0.0,
625                            f64::from(z) * #xz_scale + dz,
626                        )
627                    }}
628                } else {
629                    quote! {{
630                        let dx = #dx;
631                        let dy = #dy;
632                        let dz = #dz;
633                        noises.#field.get_value(
634                            f64::from(x) * #xz_scale + dx,
635                            f64::from(y) * #y_scale + dy,
636                            f64::from(z) * #xz_scale + dz,
637                        )
638                    }}
639                }
640            }
641
642            DensityFunction::ShiftA(s) => {
643                let field = noise_field_ident(&s.noise_id);
644                quote! { noises.#field.get_value(f64::from(x) * 0.25, 0.0, f64::from(z) * 0.25) * 4.0 }
645            }
646
647            DensityFunction::ShiftB(s) => {
648                let field = noise_field_ident(&s.noise_id);
649                quote! { noises.#field.get_value(f64::from(z) * 0.25, f64::from(x) * 0.25, 0.0) * 4.0 }
650            }
651
652            DensityFunction::Shift(s) => {
653                let field = noise_field_ident(&s.noise_id);
654                if is_flat {
655                    quote! { noises.#field.get_value(f64::from(x) * 0.25, 0.0, f64::from(z) * 0.25) * 4.0 }
656                } else {
657                    quote! { noises.#field.get_value(f64::from(x) * 0.25, f64::from(y) * 0.25, f64::from(z) * 0.25) * 4.0 }
658                }
659            }
660
661            DensityFunction::TwoArgumentSimple(t) => {
662                let a = self.gen_expr(&t.argument1, input, is_flat);
663                let b = self.gen_expr(&t.argument2, input, is_flat);
664                match t.op {
665                    TwoArgType::Add => quote! { ((#a) + (#b)) },
666                    TwoArgType::Mul => quote! { ((#a) * (#b)) },
667                    TwoArgType::Min => quote! { f64::min(#a, #b) },
668                    TwoArgType::Max => quote! { f64::max(#a, #b) },
669                }
670            }
671
672            DensityFunction::Mapped(m) => {
673                let v = self.gen_expr(&m.input, input, is_flat);
674                match m.op {
675                    MappedType::Abs => quote! { (#v).abs() },
676                    MappedType::Square => quote! { { let v = #v; v * v } },
677                    MappedType::Cube => quote! { { let v = #v; v * v * v } },
678                    MappedType::HalfNegative => {
679                        quote! { { let v = #v; if v > 0.0 { v } else { v * 0.5 } } }
680                    }
681                    MappedType::QuarterNegative => {
682                        quote! { { let v = #v; if v > 0.0 { v } else { v * 0.25 } } }
683                    }
684                    MappedType::Invert => quote! { (1.0 / (#v)) },
685                    MappedType::Squeeze => {
686                        quote! { { let c = clamp(#v, -1.0, 1.0); c / 2.0 - c * c * c / 24.0 } }
687                    }
688                }
689            }
690
691            DensityFunction::Clamp(c) => {
692                let inner = self.gen_expr(&c.input, input, is_flat);
693                let min = Literal::f64_unsuffixed(c.min);
694                let max = Literal::f64_unsuffixed(c.max);
695                quote! { clamp(#inner, #min, #max) }
696            }
697
698            DensityFunction::RangeChoice(rc) => {
699                let input_expr = self.gen_expr(&rc.input, input, is_flat);
700                let in_range = self.gen_expr(&rc.when_in_range, input, is_flat);
701                let out_range = self.gen_expr(&rc.when_out_of_range, input, is_flat);
702                let min = Literal::f64_unsuffixed(rc.min_inclusive);
703                let max = Literal::f64_unsuffixed(rc.max_exclusive);
704                quote! {{
705                    let v = #input_expr;
706                    if v >= #min && v < #max { #in_range } else { #out_range }
707                }}
708            }
709
710            DensityFunction::Spline(s) => self.gen_spline_expr(&s.spline, input, is_flat),
711
712            // TODO: BlendedNoise requires three PerlinNoise generators (min_limit, max_limit, main)
713            // with cell-based trilinear interpolation. This single-noise approximation produces
714            // incorrect terrain shapes. Implement full BlendedNoise before terrain generation.
715            DensityFunction::BlendedNoise(bn) => {
716                let field = noise_field_ident("minecraft:offset");
717                let xz = Literal::f64_unsuffixed(bn.xz_scale / bn.xz_factor);
718                let ys = Literal::f64_unsuffixed(bn.y_scale / bn.y_factor);
719                let smear = Literal::f64_unsuffixed(bn.smear_scale_multiplier);
720                quote! { noises.#field.get_value(f64::from(x) * #xz, f64::from(y) * #ys, f64::from(z) * #xz) * #smear }
721            }
722
723            DensityFunction::WeirdScaledSampler(ws) => {
724                let input_expr = self.gen_expr(&ws.input, input, is_flat);
725                let field = noise_field_ident(&ws.noise_id);
726                let mapper = match ws.rarity_value_mapper {
727                    RarityValueMapper::Tunnels => quote! { RarityValueMapper::Tunnels },
728                    RarityValueMapper::Caves => quote! { RarityValueMapper::Caves },
729                };
730                quote! {{
731                    let rarity = #input_expr;
732                    let scale = #mapper.get_values(rarity);
733                    scale * noises.#field.get_value(
734                        f64::from(x) / scale, f64::from(y) / scale, f64::from(z) / scale,
735                    ).abs()
736                }}
737            }
738
739            DensityFunction::BlendAlpha(_) => quote! { 1.0 },
740            // TODO: Transpile EndIslands properly if End terrain is ever transpiled.
741            // Currently only used via direct `EndIslands` in `EndBiomeSource`.
742            DensityFunction::BlendOffset(_) | DensityFunction::EndIslands => quote! { 0.0 },
743            DensityFunction::BlendDensity(bd) => self.gen_expr(&bd.input, input, is_flat),
744            DensityFunction::Marker(m) => self.gen_expr(&m.wrapped, input, is_flat),
745
746            DensityFunction::Reference(r) => {
747                if self.flat_cached.contains(&r.id) {
748                    // Flat-cached references are always read from the column cache
749                    let field = named_fn_field_ident(&r.id);
750                    quote! { cache.#field }
751                } else {
752                    // 3D named function — call it
753                    let fn_name = named_fn_ident(&r.id);
754                    quote! { #fn_name(noises, cache, x, y, z) }
755                }
756            }
757        }
758    }

760    /// Generate a spline evaluation expression.
761    fn gen_spline_expr(
762        &mut self,
763        spline: &CubicSpline,
764        input: &TranspilerInput,
765        is_flat: bool,
766    ) -> TokenStream {
767        let coord = self.gen_expr(&spline.coordinate, input, is_flat);
768        let n_points = spline.points.len();
769        let n_lit = Literal::usize_unsuffixed(n_points);
770
771        let locations: Vec<Literal> = spline
772            .points
773            .iter()
774            .map(|p| Literal::f32_unsuffixed(p.location))
775            .collect();
776        let derivatives: Vec<Literal> = spline
777            .points
778            .iter()
779            .map(|p| Literal::f32_unsuffixed(p.derivative))
780            .collect();
781
782        // Generate value expressions for each point
783        let value_arms: Vec<TokenStream> = spline
784            .points
785            .iter()
786            .enumerate()
787            .map(|(i, p)| {
788                let idx = Literal::usize_unsuffixed(i);
789                let val_expr = match &p.value {
790                    SplineValue::Constant(c) => {
791                        let lit = Literal::f32_unsuffixed(*c);
792                        quote! { #lit }
793                    }
794                    SplineValue::Spline(nested) => {
795                        let helper = self.gen_spline_helper(nested, input, is_flat);
796                        if is_flat {
797                            quote! { #helper(noises, cache, x, z) }
798                        } else {
799                            quote! { #helper(noises, cache, x, y, z) }
800                        }
801                    }
802                };
803                quote! { #idx => #val_expr }
804            })
805            .collect();
806
807        quote! {{
808            const LOCATIONS: [f32; #n_lit] = [#(#locations),*];
809            const DERIVATIVES: [f32; #n_lit] = [#(#derivatives),*];
810            let coord = (#coord) as f32;
811            f64::from(spline_eval::evaluate_spline(&LOCATIONS, &DERIVATIVES, coord, |__i| {
812                match __i {
813                    #(#value_arms,)*
814                    _ => unreachable!()
815                }
816            }))
817        }}
818    }
819
820    /// Generate a helper function for a nested spline, returning its ident.
821    fn gen_spline_helper(
822        &mut self,
823        spline: &Arc<CubicSpline>,
824        input: &TranspilerInput,
825        is_flat: bool,
826    ) -> Ident {
827        let id = self.spline_counter;
828        self.spline_counter += 1;
829        let fn_name = format_ident!("spline_helper_{}", id);
830
831        let body = self.gen_spline_expr(spline, input, is_flat);
832
833        let params = self.fn_params(is_flat);
834
835        self.spline_fns.push(quote! {
836            #[inline]
837            fn #fn_name(#params) -> f32 {
838                (#body) as f32
839            }
840        });
841
842        fn_name
843    }
844}

848/// Check if a density function subtree directly uses the `y` coordinate.
849/// Does NOT recurse into References (those are handled by the flat inference loop).
850fn uses_y(df: &DensityFunction) -> bool {
851    match df {
852        // uses y * 0.25
853        DensityFunction::YClampedGradient(_)
854        | DensityFunction::Shift(_)
855        | DensityFunction::BlendedNoise(_) => true,
856        DensityFunction::Noise(n) => n.y_scale != 0.0,
857        DensityFunction::ShiftedNoise(sn) => sn.y_scale != 0.0 || uses_y(&sn.shift_y),
858        DensityFunction::WeirdScaledSampler(ws) => uses_y(&ws.input),
859        DensityFunction::TwoArgumentSimple(t) => uses_y(&t.argument1) || uses_y(&t.argument2),
860        DensityFunction::Mapped(m) => uses_y(&m.input),
861        DensityFunction::Clamp(c) => uses_y(&c.input),
862        DensityFunction::RangeChoice(rc) => {
863            uses_y(&rc.input) || uses_y(&rc.when_in_range) || uses_y(&rc.when_out_of_range)
864        }
865        DensityFunction::BlendDensity(bd) => uses_y(&bd.input),
866        DensityFunction::Marker(m) => uses_y(&m.wrapped),
867        DensityFunction::Spline(s) => uses_y_spline(&s.spline),
868        // References are handled at the analysis level, not here.
869        // Constants and shift-a/b/blend/end-islands don't use y.
870        DensityFunction::Reference(_)
871        | DensityFunction::Constant(_)
872        | DensityFunction::ShiftA(_)
873        | DensityFunction::ShiftB(_)
874        | DensityFunction::BlendAlpha(_)
875        | DensityFunction::BlendOffset(_)
876        | DensityFunction::EndIslands => false,
877    }
878}
879
880fn uses_y_spline(spline: &CubicSpline) -> bool {
881    if uses_y(&spline.coordinate) {
882        return true;
883    }
884    spline.points.iter().any(|p| {
885        if let SplineValue::Spline(nested) = &p.value {
886            uses_y_spline(nested)
887        } else {
888            false
889        }
890    })
891}

907fn collect_references(df: &DensityFunction) -> Vec<String> {
908    let mut refs = Vec::new();
909    collect_refs_inner(df, &mut refs);
910    refs
911}
912
913fn collect_refs_inner(df: &DensityFunction, refs: &mut Vec<String>) {
914    match df {
915        DensityFunction::Reference(r) if !refs.contains(&r.id) => {
916            refs.push(r.id.clone());
917        }
918        DensityFunction::Marker(m) => collect_refs_inner(&m.wrapped, refs),
919        DensityFunction::TwoArgumentSimple(t) => {
920            collect_refs_inner(&t.argument1, refs);
921            collect_refs_inner(&t.argument2, refs);
922        }
923        DensityFunction::Mapped(m) => collect_refs_inner(&m.input, refs),
924        DensityFunction::Clamp(c) => collect_refs_inner(&c.input, refs),
925        DensityFunction::RangeChoice(rc) => {
926            collect_refs_inner(&rc.input, refs);
927            collect_refs_inner(&rc.when_in_range, refs);
928            collect_refs_inner(&rc.when_out_of_range, refs);
929        }
930        DensityFunction::ShiftedNoise(sn) => {
931            collect_refs_inner(&sn.shift_x, refs);
932            collect_refs_inner(&sn.shift_y, refs);
933            collect_refs_inner(&sn.shift_z, refs);
934        }
935        DensityFunction::BlendDensity(bd) => collect_refs_inner(&bd.input, refs),
936        DensityFunction::WeirdScaledSampler(ws) => collect_refs_inner(&ws.input, refs),
937        DensityFunction::Spline(s) => collect_spline_refs(&s.spline, refs),
938        _ => {}
939    }
940}
941
942fn collect_spline_refs(spline: &CubicSpline, refs: &mut Vec<String>) {
943    collect_refs_inner(&spline.coordinate, refs);
944    for point in &spline.points {
945        if let SplineValue::Spline(nested) = &point.value {
946            collect_spline_refs(nested, refs);
947        }
948    }
949}

971/// Converts a namespaced ID to a valid Rust identifier.
972///
973/// `"minecraft:overworld/continents"` → `"overworld__continents"`
974/// `"mymod:custom/noise"` → `"custom__noise"`
975fn sanitize_name(id: &str) -> String {
976    // Take just the path component, stripping any namespace (e.g. "minecraft:", "mymod:")
977    let path = match id.split_once(':') {
978        Some((_, path)) => path,
979        None => id,
980    };
981    path.replace('/', "__").replace('-', "_")
982}
