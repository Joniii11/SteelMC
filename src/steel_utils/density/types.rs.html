1//! Density function types matching vanilla Minecraft's DensityFunctions.java
2//!
3//! Each density function type is its own struct, mirroring vanilla's separate
4//! record/class pattern. The [`DensityFunction`] enum wraps them for dispatch.
5//!
6//! These types are used at build time by the density function transpiler to parse
7//! JSON density function trees and generate native Rust code. Runtime evaluation
8//! is done by the transpiled code, not by interpreting this tree.
9

312/// A density function that can be evaluated at a position to get a density value.
313///
314/// Density functions form a tree structure where complex functions are composed
315/// from simpler ones. Each variant wraps a separate struct matching vanilla's
316/// per-type class/record pattern.
317///
318/// This tree is used at build time by the transpiler to generate native Rust code.
319/// Runtime evaluation is done by the transpiled output, not by interpreting this tree.
320#[derive(Debug, Clone)]
321pub enum DensityFunction {
322    /// A constant value.
323    Constant(Constant),
324
325    /// A reference to another density function by ID.
326    Reference(Reference),
327
328    /// A Y-axis clamped gradient.
329    YClampedGradient(YClampedGradient),
330
331    /// Sample from a noise generator.
332    Noise(Noise),
333
334    /// Sample from a shifted noise generator.
335    ShiftedNoise(ShiftedNoise),
336
337    /// Shift noise generator A for coordinate offsetting.
338    ShiftA(ShiftA),
339
340    /// Shift noise generator B for coordinate offsetting.
341    ShiftB(ShiftB),
342
343    /// Generic shift noise generator for coordinate offsetting.
344    Shift(Shift),
345
346    /// Two-argument operation (add, mul, min, max).
347    TwoArgumentSimple(TwoArgumentSimple),
348
349    /// Mapped (pure transformer) operation (abs, square, cube, etc.).
350    Mapped(Mapped),
351
352    /// Clamp the value to a range.
353    Clamp(Clamp),
354
355    /// Choose between two functions based on input range.
356    RangeChoice(RangeChoice),
357
358    /// Cubic spline evaluation.
359    Spline(Spline),
360
361    /// Blended (interpolated) 3D noise.
362    BlendedNoise(BlendedNoise),
363
364    /// Weird scaled sampler (for cave generation).
365    WeirdScaledSampler(WeirdScaledSampler),
366
367    /// End islands density function (transpiler emits 0.0; actual algorithm
368    /// lives in `steel-core::worldgen::end_islands::EndIslands`).
369    EndIslands,
370
371    /// Blend alpha (returns 1.0, placeholder for blending).
372    BlendAlpha(BlendAlpha),
373
374    /// Blend offset (returns 0.0, placeholder for blending).
375    BlendOffset(BlendOffset),
376
377    /// Blend density (for chunk blending).
378    BlendDensity(BlendDensity),
379
380    /// Cache/marker wrapper (optimization hints).
381    Marker(Marker),
382}

384// ── Convenience constructors ────────────────────────────────────────────────
385
386impl DensityFunction {
387    /// Create a constant density function.
388    #[must_use]
389    pub const fn constant(value: f64) -> Self {
390        Self::Constant(Constant { value })
391    }
392
393    /// Create a reference density function (unresolved).
394    #[must_use]
395    pub const fn reference(id: String) -> Self {
396        Self::Reference(Reference { id, resolved: None })
397    }
398
399    /// Resolve all `Reference` nodes in this tree using the given registry,
400    /// and bake noise generators from `noises`.
401    ///
402    /// Call this once after building the full density function tree.
403    #[must_use]
404    pub fn resolve(
405        &self,
406        registry: &FxHashMap<String, Arc<DensityFunction>>,
407        noises: &FxHashMap<String, NormalNoise>,
408    ) -> Self {
409        self.resolve_inner(registry, noises)
410    }
411
412    fn resolve_inner(
413        &self,
414        registry: &FxHashMap<String, Arc<DensityFunction>>,
415        noises: &FxHashMap<String, NormalNoise>,
416    ) -> Self {
417        match self {
418            Self::Constant(_)
419            | Self::EndIslands
420            | Self::BlendAlpha(_)
421            | Self::BlendOffset(_)
422            | Self::YClampedGradient(_) => self.clone(),
423
424            Self::Reference(r) => {
425                let resolved = registry
426                    .get(&r.id)
427                    .map(|df| Arc::new(df.resolve_inner(registry, noises)));
428                Self::Reference(Reference {
429                    id: r.id.clone(),
430                    resolved,
431                })
432            }
433
434            Self::Noise(n) => Self::Noise(Noise {
435                noise_id: n.noise_id.clone(),
436                xz_scale: n.xz_scale,
437                y_scale: n.y_scale,
438                noise: noises.get(&n.noise_id).cloned(),
439            }),
440
441            Self::ShiftedNoise(sn) => Self::ShiftedNoise(ShiftedNoise {
442                shift_x: Arc::new(sn.shift_x.resolve_inner(registry, noises)),
443                shift_y: Arc::new(sn.shift_y.resolve_inner(registry, noises)),
444                shift_z: Arc::new(sn.shift_z.resolve_inner(registry, noises)),
445                xz_scale: sn.xz_scale,
446                y_scale: sn.y_scale,
447                noise_id: sn.noise_id.clone(),
448                noise: noises.get(&sn.noise_id).cloned(),
449            }),
450
451            Self::ShiftA(s) => Self::ShiftA(ShiftA {
452                noise_id: s.noise_id.clone(),
453                noise: noises.get(&s.noise_id).cloned(),
454            }),
455
456            Self::ShiftB(s) => Self::ShiftB(ShiftB {
457                noise_id: s.noise_id.clone(),
458                noise: noises.get(&s.noise_id).cloned(),
459            }),
460
461            Self::Shift(s) => Self::Shift(Shift {
462                noise_id: s.noise_id.clone(),
463                noise: noises.get(&s.noise_id).cloned(),
464            }),
465
466            Self::TwoArgumentSimple(t) => Self::TwoArgumentSimple(TwoArgumentSimple {
467                op: t.op,
468                argument1: Arc::new(t.argument1.resolve_inner(registry, noises)),
469                argument2: Arc::new(t.argument2.resolve_inner(registry, noises)),
470            }),
471
472            Self::Mapped(m) => Self::Mapped(Mapped {
473                op: m.op,
474                input: Arc::new(m.input.resolve_inner(registry, noises)),
475            }),
476
477            Self::Clamp(c) => Self::Clamp(Clamp {
478                input: Arc::new(c.input.resolve_inner(registry, noises)),
479                min: c.min,
480                max: c.max,
481            }),
482
483            Self::RangeChoice(rc) => Self::RangeChoice(RangeChoice {
484                input: Arc::new(rc.input.resolve_inner(registry, noises)),
485                min_inclusive: rc.min_inclusive,
486                max_exclusive: rc.max_exclusive,
487                when_in_range: Arc::new(rc.when_in_range.resolve_inner(registry, noises)),
488                when_out_of_range: Arc::new(rc.when_out_of_range.resolve_inner(registry, noises)),
489            }),
490
491            Self::Spline(s) => Self::Spline(Spline {
492                spline: Arc::new(resolve_spline(&s.spline, registry, noises)),
493            }),
494
495            Self::BlendedNoise(bn) => Self::BlendedNoise(BlendedNoise {
496                xz_scale: bn.xz_scale,
497                y_scale: bn.y_scale,
498                xz_factor: bn.xz_factor,
499                y_factor: bn.y_factor,
500                smear_scale_multiplier: bn.smear_scale_multiplier,
501                noise: noises.get("minecraft:offset").cloned(),
502            }),
503
504            Self::WeirdScaledSampler(ws) => Self::WeirdScaledSampler(WeirdScaledSampler {
505                input: Arc::new(ws.input.resolve_inner(registry, noises)),
506                noise_id: ws.noise_id.clone(),
507                rarity_value_mapper: ws.rarity_value_mapper,
508                noise: noises.get(&ws.noise_id).cloned(),
509            }),
510
511            Self::BlendDensity(bd) => Self::BlendDensity(BlendDensity {
512                input: Arc::new(bd.input.resolve_inner(registry, noises)),
513            }),
514
515            Self::Marker(m) => Self::Marker(Marker {
516                kind: m.kind,
517                wrapped: Arc::new(m.wrapped.resolve_inner(registry, noises)),
518            }),
519        }
520    }
521}

523/// Resolve noise/registry references within a cubic spline.
524fn resolve_spline(
525    spline: &CubicSpline,
526    registry: &FxHashMap<String, Arc<DensityFunction>>,
527    noises: &FxHashMap<String, NormalNoise>,
528) -> CubicSpline {
529    let points: Vec<SplinePoint> = spline
530        .points
531        .iter()
532        .map(|p| SplinePoint {
533            location: p.location,
534            value: match &p.value {
535                SplineValue::Constant(v) => SplineValue::Constant(*v),
536                SplineValue::Spline(nested) => {
537                    SplineValue::Spline(Arc::new(resolve_spline(nested, registry, noises)))
538                }
539            },
540            derivative: p.derivative,
541        })
542        .collect();
543    CubicSpline::new(
544        Arc::new(spline.coordinate.resolve_inner(registry, noises)),
545        points,
546    )
547}

571/// Rarity value mapper for cave generation.
572///
573/// Used at runtime by transpiled `WeirdScaledSampler` code.
574#[derive(Debug, Clone, Copy, PartialEq, Eq)]
575pub enum RarityValueMapper {
576    /// Mapper type `"type_1"` for tunnels.
577    Tunnels,
578    /// Mapper type `"type_2"` for caves.
579    Caves,
580}
581
582impl RarityValueMapper {
583    /// Get the scaling factor for this mapper based on rarity value.
584    ///
585    /// From vanilla NoiseRouterData.QuantizedSpaghettiRarity.
586    #[must_use]
587    pub fn get_values(&self, rarity: f64) -> f64 {
588        match self {
589            // Type 1: getSpaghettiRarity3D (tunnels)
590            Self::Tunnels => {
591                if rarity < -0.5 {
592                    0.75
593                } else if rarity < 0.0 {
594                    1.0
595                } else if rarity < 0.5 {
596                    1.5
597                } else {
598                    2.0
599                }
600            }
601            // Type 2: getSpaghettiRarity2D (caves)
602            Self::Caves => {
603                if rarity < -0.75 {
604                    0.5
605                } else if rarity < -0.5 {
606                    0.75
607                } else if rarity < 0.5 {
608                    1.0
609                } else if rarity < 0.75 {
610                    2.0
611                } else {
612                    3.0
613                }
614            }
615        }
616    }
617}

619/// A cubic spline for density function interpolation.
620#[derive(Debug, Clone)]
621pub struct CubicSpline {
622    /// The coordinate extractor (which density function to use as input)
623    pub coordinate: Arc<DensityFunction>,
624    /// The spline points
625    pub points: Vec<SplinePoint>,
626    /// Pre-extracted point locations for binary search (avoids allocation per eval).
627    pub locations: Vec<f32>,
628}
629
630/// A point in a cubic spline.
631#[derive(Debug, Clone)]
632pub struct SplinePoint {
633    /// The location (input value) of this point.
634    pub location: f32,
635    /// The value or nested spline at this point.
636    pub value: SplineValue,
637    /// The derivative at this point.
638    pub derivative: f32,
639}
640
641/// A spline point value can be either a constant or a nested spline.
642#[derive(Debug, Clone)]
643pub enum SplineValue {
644    /// A constant value.
645    Constant(f32),
646    /// A nested spline.
647    Spline(Arc<CubicSpline>),
648}
649
650impl CubicSpline {
651    /// Create a new cubic spline.
652    #[must_use]
653    pub fn new(coordinate: Arc<DensityFunction>, points: Vec<SplinePoint>) -> Self {
654        let locations = points.iter().map(|p| p.location).collect();
655        Self {
656            coordinate,
657            points,
658            locations,
659        }
660    }
661}
